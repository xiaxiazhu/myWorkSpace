  5 KISSY.add("gallery/grid/1.0/buttonBar",function(S,bar){
  6 
  7 	var CLS_BUTTON_BAR = "button-bar";
  8 	//按钮栏
  9 	function buttonBar(config){
 10 		var _self = this;
 11 		config = _self._formatConfig(config || {});
 12 		
 13 		buttonBar.superclass.constructor.call(_self, config);
 14 		//_self._init();
 15 	}
 16 	S.extend(buttonBar, bar);
 17 
 18 	S.augment(buttonBar,{
 19 		CLS_BAR : CLS_BUTTON_BAR,
 20 		//格式化配置
 21 		_formatConfig : function (config){
 22 			var result ={renderTo : config.renderTo,items:[]},
 23 				items = config.buttons || config.items;
 24 			
 25 			S.each(items, function (button){
 26 				var item = S.merge({type:'button'},button);
 27 				result.items.push(item);
 28 			});
 29 
 30 			return result;
 31 		}
 32 	});
 33 
 34 	return buttonBar;
 35 }, {requires : ["./bar"]});
