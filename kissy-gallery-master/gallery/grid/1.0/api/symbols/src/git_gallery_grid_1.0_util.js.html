1 /** @fileOverview 对KISSY进行扩展的一些帮助函数
  2 * 包括：屏蔽层，格式化函数，Form帮助类

  6 KISSY.add("gallery/grid/1.0/util",function(S){
  7 	
  8 	var DOM = S.DOM, Event = S.Event,
  9         win = window, doc = document,UA=S.UA;
 10 
 11 	/**
 12 	* @memberOf Grid
 13 	* @name Util
 14 	* @class
 15 	*/
 16 	var Util = {};
 17 
 18 	S.mix(Util,
 19 	/** @lends  Grid.Util */
 20 	{
 21 		/**
 22 		* @description 屏蔽整个页面: <br> 1）ie 6,ie 7下设置100%无效 <br> 2) 兼容浏览器的可视区域和内容区域
 23 		* @example 
 24 		*	Grid.Util.mask();	//屏蔽窗口
 25 		*	Grid.Util.unmask();	//解除屏蔽窗口
 26 		*/
 27 		mask : function(){
 28 			var bodyEl = S.one('body'),
 29 				bodyHeight = bodyEl.height(),
 30 				viewHeight = DOM.viewportHeight(),
 31 				height = bodyHeight > viewHeight ?bodyHeight :viewHeight,
 32 				maskEl = Util.maskElement(bodyEl);
 33 			
 34 			maskEl.height(height);
 35 			Event.on(win,'resize',resizeEvent);
 36 			maskEl.data('reszieFunc',resizeEvent);
 37 			var funcId =null;
 38 			function resizeEvent(){
 39 
 40 				if(funcId)
 41 				{
 42 					clearTimeout(funcId);
 43 				}
 44 				funcId = setTimeout(setHeight,300);
 45 			}
 46 
 47 			function setHeight(){
 48 				var viewHeight = DOM.viewportHeight(),
 49 					height = bodyHeight > viewHeight ?bodyHeight :viewHeight;
 50 				maskEl.height(height);
 51 			}
 52 		},
 53 		/**
 54 			@description 取消屏蔽整个页面
 55 		*/
 56 		unmask :function(){
 57 			var bodyEl = S.one('body'),
 58 				maskEl = bodyEl.children('.lp-el-mask'),
 59 				func = maskEl.data('reszieFunc');
 60 			if(func){
 61 				Event.remove(win,'resize',func);
 62 				maskEl.data('reszieFunc',null);
 63 			}
 64 			Util.unmaskElement(bodyEl);
 65 		},
 66 		/**
 67 		* @description 屏蔽指定元素
 68 		* @param {[String|DOM|Node]} element 屏蔽的元素，可以使用选择器、Dom元素，Node元素
 69 		* @param {String} msg 屏蔽层上显示的信息，可以为空
 70 		* @param {String} msgCls 屏蔽信息上应用的CSS,可以为空，此项仅在 msg有效时起作用
 71 		* @example 
 72 		*	Grid.Util.maskElement('#domId');	//屏蔽元素，暂时只屏蔽选择器的第一个元素
 73 		*/
 74 		maskElement: function (element, msg, msgCls) {
 75 			var maskedEl = S.one(element),
 76 				maskedNode = maskedEl.getDOMNode(),
 77 				maskDiv = S.one('.lp-el-mask',maskedNode);
 78 			if (!maskDiv) {
 79 				var maskDiv = S.one(DOM.create('<div class="lp-el-mask"></div>')).appendTo(maskedNode);
 80 				maskedEl.addClass('x-masked-relative x-masked');
 81 				if(UA.ie == 6){
 82 					maskDiv.height(maskedEl.height());
 83 				}
 84 				if (msg) {
 85 					var template = ['<div class="lp-el-mask-msg"><div>', msg, '</div></div>'].join(''),
 86 						msgDiv = S.one(DOM.create(template)).appendTo(maskedNode);
 87 					if (msgCls) {
 88 						msgDiv.addClass(msgCls);
 89 					}
 90 					try{
 91 						var left = (maskedEl.width()- msgDiv.width()) / 2,
 92 							top = (maskedEl.height() - msgDiv.height()) / 2;
 93 						msgDiv.css({ left: left, top: top });
 94 					}catch(ex){
 95 						S.log('width error!');
 96 					}
 97 				}
 98 			}
 99 			return maskDiv;
100 		},
101 		/**
102 		* @description 解除对应元素的屏蔽
103 		* @param {[String|DOM|Node]} element 屏蔽的元素，可以使用选择器、Dom元素，Node元素
104 		* @example 
105 		*	Grid.Util.unmaskElement('#domId');	//解除屏蔽元素，暂时只支持选择器的第一个元素
106 		*/ 
107 		unmaskElement: function (element) {
108 			var maskedEl = S.one(element),
109 				maskedNode = maskedEl.getDOMNode(),
110 				msgEl = maskedEl.children('.lp-el-mask-msg'),
111 				maskDiv = maskedEl.children('.lp-el-mask');
112 			if(msgEl){
113 				msgEl.remove();
114 			}
115 			if(maskDiv){
116 				maskDiv.remove();
117 			}
118 			maskedEl.removeClass('x-masked-relative x-masked');
119 
120 		}
121 	});

123 	/**
124 	* 格式化数据的帮助方法
125 	* @class 格式化函数
126 	* @name Grid.Util.Format
127 	*/
128 	Util.Format = function(){
129 		/** @lends  Grid.Util.Format */	
130 		return {
131 			/**
132 				@description 日期格式化函数
133 				@param {Number|Date} date 格式话的日期，一般为1970 年 1 月 1 日至今的毫秒数 
134 				@return {String} 格式化后的日期格式为 2011-10-31
135 				@example
136 			* 一般用法：<br> 
137 			* Grid.Util.Format.dateRenderer(1320049890544);输出：2011-10-31 <br>
138 			* 表格中用于渲染列：<br>
139 			* {title:"出库日期",dataIndex:"date",renderer:Util.Format.dateRenderer}
140 			*/
141 			dateRenderer: function (d) {
142 				if(!d){
143 					 return "";
144 				}
145                 try {
146                     var date =new Date(d);// eval(d.replace(/\/Date\((\d+)\)\//gim, "new Date($1)"));
147                 } catch (e) {
148                     return "";
149                 }
150                 if (!date || !date.getFullYear)
151                     return "";
152                 return S.Date.format(d,"yyyy-mm-dd");
153             },
154 			/**
155 				@description 日期时间格式化函数
156 				@param {Number|Date} date 格式话的日期，一般为1970 年 1 月 1 日至今的毫秒数 
157 				@return {String} 格式化后的日期格式时间为 2011-10-31 16 : 41 : 02
158 			*/
159 			datetimeRenderer: function (d) {
160 				if(!d){
161 					 return "";
162 				}
163                 try {
164                     var date =new Date(d);// eval(d.replace(/\/Date\((\d+)\)\//gim, "new Date($1)"));
165                 } catch (e) {
166                     return "";
167                 }
168                 if (!date || !date.getFullYear)
169                     return "";
170 				var dateString = S.Date.format(d,"yyyy-mm-dd HH:MM:ss");
171 				//var dateString = date.getFullYear() + "-" + (date.getMonth() + 1) + "-" + date.getDate() + ' ' + date.getHours() + ":" + date.getMinutes() + ":" + date.getSeconds();
172                 return dateString;//.replace(/(\s)(\d):/,' 0$2:').replace(/:(\d):/,':0$1:').replace(/:(\d)$/,':0$1'));
173 			},
174 			/**
175 				@description 文本截取函数，当文本超出一定数字时，会截取文本，添加...
176 				@param {Number} length 截取多少字符
177 				@return {String} 返回截取后的字符串，如果本身小于指定的数字，返回原字符串。如果大于，则返回截断后的字符串，并附加...
178 			*/
179 			cutTextRenderer : function(length){
180 				return function(value){
181 					value = value || '';
182 					if(value.toString().length > length){
183 						return value.toString().substring(0,length)+"...";
184 					}
185 					return value;
186 				}
187 			},
188 			/**
189 			* @description 枚举格式化函数
190 			* @param {Object} enumObj 键值对的枚举对象 {"1":"大","2":"小"}
191 			* @return {Function} 返回指定枚举的格式化函数
192 			* @example 
193 			* //Grid 的列定义
194 			*  {title:"状态",dataIndex:"status",renderer:Grid.Util.Format.enumRenderer({"1":"入库","2":"出库"})}
195 			*/
196 			enumRenderer : function(enumObj){
197 				return function(value){
198 					return enumObj[value] || '';
199 				}
200 			},
201 			/*
202 			* @description 将多个值转换成一个字符串
203 			* @param {Object} enumObj 键值对的枚举对象 {"1":"大","2":"小"}
204 			* @return {Function} 返回指定枚举的格式化函数
205 			* @example 
206 			* //Grid 的列定义
207 			*  {title:"状态",dataIndex:"status",renderer:Grid.Util.multipleItemsRenderer({"1":"入库","2":"出库","3":"退货"})}
208 			*  //数据源是[1,2] 时，则返回 "入库,出库"
209 			*/
210 			multipleItemsRenderer : function(enumObj){
211 				var enumFun = Util.Format.enumRenderer(enumObj);
212 				return function(values){
213 					if(!values)
214 						return '';
215 					var result = [];
216 					if(S.isArray(values)){
217 						values = values.toString().split(',');
218 					}
219 					S.each(values,function(value){
220 						result.push(enumFun(value));
221 					});
222 					
223 					return result.join(',');
224 				}
225 			}
226 		};
227 	}();

230 	* 表单的一些工具方法
231 	* @class 格式化函数
232 	* @name  Grid.Util.FormHelpr
233 	*/
234 	Util.FormHelpr = 
235 	/** @lends  Grid.Util.FormHelpr */		
236 	{
237 		/**
238 		* 将表单数据序列化成为字符串
239 		* @param {HTMLForm} form 表单元素
240 		* @return {String} 序列化的字符串
241 		*/
242 		serialize:function(form){
243 			return S.param(Util.FormHelpr.serializeToObject(form));
244 		},
245 		/**
246 		* 将表单数据序列化成对象
247 		* @param {HTMLForm} form 表单元素
248 		* @return {Object} 表单元素的
249 		*/
250 		serializeToObject:function(form){
251 			var elements = S.makeArray(form.elements);
252 			elements = S.filter(elements,function(item){
253 				return (item.id ||item.name) && !item.disabled &&
254 					(this.checked || /select|textarea/i.test(item.nodeName) ||
255 						/text|hidden|password/i.test(item.type));
256 			});
257 			var arr =[];
258 			S.each(elements,function(elem){
259 				var val = S.one(elem).val(),
260 					obj = val == null ? null : S.isArray(val) ?
261 					S.map( val, function(val, i){
262 							return {name: elem.name||elem.id, value: val};
263 					}) :
264 					{name:  elem.name||elem.id, value: val};
265 				if(obj){
266 					arr.push(obj);
267 				}
268 			});
269 			var result={};
270 			S.each(arr,function(elem){
271 				var prop = result[elem.name];
272 				if(!prop){
273 					result[elem.name] = elem.value;
274 				}else if(S.isArray(prop)){
275 					prop.push(elem.value);
276 				}else{
277 					var a = [];
278 					a.push(prop);
279 					a.push(elem.value);
280 					result[elem.name]=a;
281 				}
282 			});
283 			return result;
284 		}
285 	};
286 		
287 	return Util;
