287 		* @example 
288 		*	grid.filter('id',124); //仅显示 id ==124的行

290 		*	grid.filter('id',function(value){
291 		*		if(value>124)

297 			var _self = this,
298 				body = _self.get('tbody'),
299 				rows = S.makeArray(body.rows),
300 				func = typeof value === 'function' ? value : function (val) {return val === value; };

302 				var rowEl = S.one(row),
303 					obj = DOM.data(row, DATA_ELEMENT);
304 				if (value === null) {
305 					rowEl.show();
306 				} else if (!obj || !func(obj[field])) {
307 					rowEl.hide();
308 				} else {
309 					rowEl.show();
310 				}
311 			});

530 		_findRowByRecord : function (record) {
531 			var _self = this,
532 				tbody = _self.get('tbody'),
533 				rows = tbody.rows,
534 				result = null;
535             S.each(rows, function (row) {
