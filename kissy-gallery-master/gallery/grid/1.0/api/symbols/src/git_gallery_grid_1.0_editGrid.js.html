  3 		UA = S.UA,

  6 	var ATTR_COLUMN_NAME = 'data-column-name',
  7 		DATA_ELEMENT = 'row-element',
  8 		CLS_CELL_TEXT = 'grid-body-cell-text',
  9 		CLS_GRID_CELL = 'grid-body-cell',
 10 		CLS_GRID_CELL_INNER = 'grid-body-cell-inner',
 11 		KEY_COLUMN_PREFIX = 'col',
 12 		CLS_EDITOR_FIELD = 'lp-editor-field',
 13 		CLS_CELL_EDITABLE = 'grid-editable-cell',
 14 		CLS_CELL_ERROR = 'grid-error-cell',
 15 		CLS_CELL_ERROR_ICON = 'grid-cell-error-icon',
 16 		PREFIX_CLS_CELL = 'grid-body-td-';

 18 	/**
 19 	* 可编辑表格控件
 20 	* @memberOf Grid
 21 	* @name EidtGrid
 22 	* @description 编辑表格数据
 23 	* @class 可编辑的表格类
 24 	* @augments Grid.Grid 
 25 	* @param {Object} conifg 配置项,同父类{@link KISSY.LP.Grid},差别在于列的配置上
 26 	* @param {Object} conifg.showError 是否直接在列表中显示错误提示
 27 	* @param {Object} conifg.columns[0].editor 列上的编辑器配置<br>
 28 	* 1) type 编辑器的类型，目前支持 "text","number","select"<br>
 29 	* 2) items 当类型为 "select"时，items中包含下拉列表中的键值对<br>
 30 	* 3) validator ,函数原型 Function(value,obj)验证器，当返回值为字符串时，将其显示成错误信息<br>
 31 	* 4) editableFun 函数原型 Function(value,obj) 决定当前单元格是否可以编辑，如果返回值为true 则可以编辑。否则不可以编辑
 32 	*  默认值为返回 function(){return true};
 33 	* @example 
 34 	* //数字编辑器，并附加验证
 35 	* { title: '总花费', sortable: true,  dataIndex: 'TotalCost',editor:{type:'number',validator:function(value,obj){
 36 	*		if(value > 100){
 37 	*			return '总花费不大于100';
 38 	*		}
 39 	*	}}
 40 	* }
 41 	* //下拉列表编辑器
 42 	* { title: '选择', sortable: true,  dataIndex: 'check',editor:{type:'select',items:[{name:'选择1',value:'1'},{name:'选择2',value:'2'}]}}
 43 	* @see Grid.Grid
 44 	*/
 45 	function EditGrid(config) {
 46 		EditGrid.superclass.constructor.call(this, config);
 47 	}
 48 
 49 	S.extend(EditGrid, Grid, 
 50 	/** @lends Grid.EidtGrid.prototype */	
 51 	{
 52 		/**
 53 		* 清除错误，如果还处于编辑状态，则取消编辑状态
 54 		*/
 55 		clearError : function(){
 56 			var _self = this,
 57 				result = false,
 58 				showError = _self.get('showError'),
 59 				columns = null,
 60 				tbody = null;
 61 			if(showError){
 62 				/*tbody = _self.get('tbody');
 63 				S.one(tbody).all('.' + CLS_CELL_ERROR).removeClass(CLS_CELL_ERROR);*/
 64 			}
 65 			_self.cancelEdit();
 66 		},
 67 		/**
 68 		* 取消编辑状态，隐藏所有的编辑器，无论是否当前编辑器内的数据通过验证
 69 		*/
 70 		cancelEdit : function(){
 71 			var _self = this,
 72 				columns = _self.get('columns');
 73 			S.each(columns,function(column){
 74 					var editor = column.editor;
 75 				if(editor && !editor.isHide()){
 76 					editor.hide();
 77 				}
 78 			});
 79 		},
 80 		/**
 81 		* 是否包含错误
 82 		*/
 83 		hasError : function(){
 84 			var _self = this,
 85 				result = false,
 86 				showError = _self.get('showError'),
 87 				columns = null,
 88 				tbody = null;
 89 			if(showError){
 90 				tbody = _self.get('tbody');
 91 				result = !! S.one(tbody).one('.' + CLS_CELL_ERROR);
 92 			}
 93 			if(!result){
 94 				columns = _self.get('columns')
 95 				S.each(columns,function(column){
 96 					var editor = column.editor;
 97 					if(editor && !editor.isHide() && editor.hasError()){
 98 						result = true;
 99 						return false;
100 					}
101 				});
102 			}
103 			return result;
104 		},
105 		/**
106 		* 设置单元格进入编辑状态
107 		* @param {Object} record 编辑的记录
108 		* @param {String} field 编辑的字段
109 		*/
110 		setCellToEdit : function(record,field){
111 			var _self = this,
112 				cell = _self._getCell(record,field),
113 				editor = _self._getEditor(field);
114 			if(DOM.hasClass(cell,CLS_CELL_EDITABLE)){
115 				_self._showEditor(editor,cell, record[field],record);
116 			}
117 		},
118 		//初始化编辑事件
119 		_attachEditEvent : function(column){
120 			var _self = this,
121 				editor = column.editor;
122 			editor.on('change',_self._getChangeEvent(column));
123 
124 			editor.on('hide',function(event){
125 				_self.set('editRecord',null);
126 				_self.set('editor',null);
127 			});
128 		},
129 		//数据改变触发的事件
130 		_getChangeEvent : function(column){
131 			var _self = this;
132 			return function(event){
133 				var value = event.value,
134 					record = event.record,
135 					field = column.dataIndex,
136 					store = _self.get('store');
137 				if(record[field] !== value){
138 					store.setValue(record, field, value);
139 				}
140 			}
141 		},
142 		//展示错误信息
143 		_addError : function (cell, msg) {
144 			var innerEl = cell.one('.' + CLS_GRID_CELL_INNER),
145 				errorEl = cell.one('.' + CLS_CELL_ERROR_ICON),
146 				temp = ['<span class="', CLS_CELL_ERROR_ICON, '" title="', msg, '"></span>'].join('');
147 			if(!errorEl){
148 				innerEl.addClass(CLS_CELL_ERROR);
149 				errorEl = new Node(temp).appendTo(innerEl);
150 			}else{
151 				errorEl.attr('title',msg);
152 			}
153 		},
154 		//清除错误
155 		_clearError : function(cell){
156 			var innerEl = cell.one('.' + CLS_GRID_CELL_INNER),
157 				errorEl = cell.one('.' + CLS_CELL_ERROR_ICON);
158 			innerEl.removeClass(CLS_CELL_ERROR);
159 			if(errorEl){
160 				errorEl.remove();
161 			}
162 		},
163 		//获取编辑器
164 		_getEditor : function (field) {
165 			var _self = this
166 				columns = _self.get('columns'),
167 				editor = null;
168 			S.each(columns, function(column){
169 				if(column.dataIndex === field){
170 					editor = column.editor; 
171 					return false;
172 				}
173 			});
174 			return editor;
175 		},
176 		//获取单元格的模版/**/
177 		_getCellTemplate : function (colindex, column, text,value,obj) {
178 			var _self = this,
179 				dataIndex = column.dataIndex,
180 				width = column.width,
181 				tipText = column.showTip ? 'title = "' + (value||'') + '"' : '',
182 				hideText = column.hide ? 'ks-hidden' : '',
183 				editor = column.editor,
184 				eitable = editor ? (editor.editableFun && !editor.editableFun(value,obj) ? '' : CLS_CELL_EDITABLE) : '',
185 				showError = _self.get('showError'),
186 				validText = showError ? (editor && editor.validator ? editor.validator(value,obj) : ''): '',
187 				errorText = validText ?  '<span class="' + CLS_CELL_ERROR_ICON + '" title="' + validText + '"></span>' : '',
188 				errorCls =  validText ? CLS_CELL_ERROR : '',
189 				template = ['<td  class="grid-body-cell grid-body-td-', dataIndex, ' ', hideText, ' ', eitable ,'" data-column-name="', dataIndex, '" colindex="', colindex, '" width="', width, 'px">',
190 						'<div class="',CLS_GRID_CELL_INNER,' ', errorCls ,'" style="width : ', width, 'px"><span class="', CLS_CELL_TEXT, ' " ' , tipText, '>', text, '</span>', errorText, '</div></td>'].join('');
191 			return template;
192 		},
193 		//获取编辑器模版
194 		_getTextEditorTemplate : function (width) {
195 			width = width || 80;
196 			var temp = ['<div class="lp-text-editor grid-editor" style="position : absolute; z-index : 1000; visibility : hidden; left : -10000px; top : -10000px; overflow : auto;">',
197 				'<input type="text" name=""autocomplete="off" size="20" class="lp-form-text-field lp-editor-field" style="width : ', width, 'px; height : 20px;">'].join('');
198 			return temp;
199 		},
200 		//获取行数据
201 		_getRowDataByCell : function (cell){
202 			var _self = this,
203 				row = _self._findRow(cell),
204 				obj = DOM.data(row, DATA_ELEMENT);
205 			return obj;
206 		},
207 		//获取单元格
208 		_getCell : function(record,field){
209 			var _self = this,
210 				row = _self._findRowByRecord(record),
211 				cls = PREFIX_CLS_CELL + field,
212 				cell = S.one('.' + cls, row);
213 			return cell;
214 		},
215 		//获取单元格上文本信息
216 		_getCellValue : function (cell, field) {
217 			var _self = this,
218 				row = _self._findRow(cell),
219 				obj = DOM.data(row, DATA_ELEMENT),
220 				value = obj ? obj[field] : null;
221 			return value;
222 		},
223 		_hasEditor : function (field) {
224 			return this.hasAttr(KEY_COLUMN_PREFIX + field);
225 		},
226 		_hideEditor : function (editor) {
227 			editor.css({visibility : 'hidden', left : -10000, top : -10000});
228 		},
229 		//初始化
230 		_init : function () {
231 			EditGrid.superclass._init.call(this);
232 			var _self = this,
233 				columns = _self.get('columns');
234 			S.each(columns, function (column) {
235 				var editor = column.editor,
236 					cls = null;
237 				if (!editor || editor.isEditor) {
238 					return;
239 				}
240 				cls = GridEditor.types[editor.type||'text'];
241 				column.editor = new cls(editor);
242 				_self._attachEditEvent(column);
243 				//_self._createEitor(column);
244 			});
245 		},
246 		_initEvent : function(){
247 			var _self = this;
248 			S.Event.on(document,'click',function(event){
249 				var editor = _self.get('editor'),
250 					sender = event.target,
251 					cell = _self._findCell(event.target);
252 				if(!cell && editor && !editor.isHide() && !editor.hasError() && !editor.isContains(sender)){
253 					editor.hide();
254 				}
255 			});/**/
256 			_self.constructor.superclass._initEvent.call(_self);
257 		},
258 		_initDataEvent : function () {
259 
260 			var _self = this,
261 				store = _self.get('store');
262 			store.on('updaterecord', function (event) {
263 				var record = event.record;
264 				_self._setRowValue(record);
265 			});
266 			_self.on('rowremoved',function(event){
267 				var record = _self.get('editRecord'),
268 					editor = _self.get('editor');
269 				if(record && editor){
270 					if(record == event.data && !editor.isHide()){
271 						editor.hide();
272 					}
273 				}
274 			});
275 			_self.constructor.superclass._initDataEvent.call(this);
276 		},
277 		//点击行事件，决定是否显示编辑器
278 		_rowClickEvent : function (target) {
279 			var _self = this,
280 				cell = _self._findCell(target),
281 				field = null,
282 				editor = null,
283 				formerEditor = null,
284 				record = null;
285 			if (cell) {
286 				field = DOM.attr(cell, ATTR_COLUMN_NAME);
287 				editor = _self._getEditor(field);
288 				formerEditor = _self.get('editor');
289 				if(formerEditor && formerEditor != editor){
290 					if(!formerEditor.isHide() && !formerEditor.hasError()){
291 						formerEditor.hide();
292 					}
293 				}
294 				if (field && editor) {
295 					
296 					if(!editor.isHide() && editor.hasError()){
297 						editor._focus();
298 						return;
299 					}
300 					record = _self._getRowDataByCell(cell);
301 					if(record){
302 						//验证是否可以编辑，如果不可编辑则不显示编辑器
303 						if(!DOM.hasClass(cell,CLS_CELL_EDITABLE))
304 						{
305 							return ;
306 						}
307 						var result =_self.fire('beforeedit',{cell : cell , field : field , data : record});
308 						if(result !== false){
309 							
310 							_self._showEditor(editor,cell, record[field],record);
311 						}
312 					}
313 					return;
314 				}
315 			}
316 			_self.constructor.superclass._rowClickEvent.call(this, target);
317 		},
318 		_setRowValue : function (obj) {
319 			var _self = this,
320 				row = _self._findRowByRecord(obj),
321 				columns = _self.get('columns');
322 			if (row) {
323 				S.each(columns, function (column) {
324 					var field = column.dataIndex,
325 						cls = PREFIX_CLS_CELL + field,
326 						cell = S.one('.' + cls, row),
327 						textEl = cell.one('.' + CLS_CELL_TEXT),
328 						editor = column.editor,
329 						validText = editor && editor.validator ? editor.validator(obj[field],obj) : '',
330 						text = null,
331 						showError = _self.get('showError');
332 					if (textEl) {
333 						text =  column.renderer ? column.renderer(obj[field], obj) : obj[field];
334 						text = text === undefined ? '' : text;
335 						textEl.html(text);
336 						if(!showError){
337 							return;
338 						}
339 						if(validText){
340 							_self._addError(cell,validText);
341 						}else{
342 							_self._clearError(cell);
343 						}
344 					}
345 				});
346 			}
347 		},
348 		_showEditor : function (eidtor,cell, value,record) {
349 			var _self = this,
350 				offset = DOM.offset(cell),
351 				width = DOM.width(cell),
352 				inner = DOM.get('.' + CLS_GRID_CELL_INNER,cell),
353 				height = DOM.height(inner),
354 				preEidtor = _self.get('editor');
355 				
356 			
357 			if(preEidtor && !preEidtor.isHide() && !preEidtor.hasError()){
358 				preEidtor.hide();
359 			}
360 			_self.set('editRecord',record);
361 			_self.set('editor',editor);
362 			if(UA.ie == 7){
363 				eidtor.setWidth(width - 4);
364 			}else{
365 				eidtor.setWidth(width - 2);
366 			}
367 			//eidtor.setHeight(height - 2);
368 			eidtor.setValue(value,record);
369 			//eidtor.show(offset);
370 			eidtor.show({container:inner,left:0,top:2});
371 		}
372 	});
