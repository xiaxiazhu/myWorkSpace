  2 KISSY.add("gallery/grid/1.0/calendar",function(S){
  3 	function calendar(list){
  4 		var DOM = S.DOM 
  5 			Event = S.Event,
  6 			date = {} ;
  7 		S.each(list,function(item){
  8 			var node = DOM.get(item.selector);
  9 			if(!node) return ;
 10 			
 11 			//给时间控件添加小按钮，设置长度,取消只读
 12 			var timeInput = S.one(item.selector);
 13 			if(timeInput.hasAttr('readonly')){
 14 				timeInput.removeAttr('readonly');
 15 			}
 16 			if(!timeInput.hasClass('ks-select-calendar')){
 17 				timeInput.addClass('ks-select-calendar');
 18 			}
 19 			if(item.showTime){
 20 				if(!timeInput.hasClass('calendar-time')){
 21 					timeInput.addClass('calendar-time');
 22 				}
 23 			}
 24 			var datatime ,
 25 				method = item.showTime ? "timeSelect" : "select" ,
 26 				format = item.showTime ? "yyyy-mm-dd HH:MM:ss" : "yyyy-mm-dd",
 27 				min = item.min ? showdate(item.min, new Date()) : (item.min == 0 ? new Date() : null ) ,
 28 				max = item.max ? showdate(item.max, new Date()) : (item.max == 0 ? new Date() : null );
 29 				
 30 			if(item.config){
 31 				datatime = new S.Calendar(item.selector,item.config) ;
 32 			}
 33 			else{
 34 				var selected = Date.parse(DOM.val(item.selector).replace(/\-/g,"/"));
 35 				datatime = new S.Calendar(item.selector,{
 36 					minDate:min,
 37 					maxDate:max,
 38 					selected:selected ? new Date(selected) : null ,
 39 					showTime:item.showTime,
 40 					popup:true,
 41 					triggerType:['click']
 42 				});
 43 			}
 44 			datatime.on(method,function(ev){
 45 				var p = DOM.get(item.selector) ;
 46 				DOM.val(p,getDate(ev.date)) ;
 47 				this.hide();
 48 				p.focus();
 49 			});
 50 
 51 			Event.on(item.selector,"valuechange",function(ev){
 52 				if(!(ev.keyCode == 8 || ev.keyCode == 46)){
 53 					DOM.val(this,"");
 54 				}
 55 			});
 56 
 57 			function getDate(date){
 58 				return S.Date.format(date,format)
 59 			}
 60 
 61 			date[item.selector] = datatime ;
 62 		});
 63 
 64 		return date;
 65 	}
 66 
 67 	var showdate = function(n, d) {//计算d天的前几天或者后几天，返回date,注：chrome下不支持date构造时的天溢出
 68         var uom = new Date(d - 0 + n * 86400000);
 69         uom = uom.getFullYear() + "/" + (uom.getMonth() + 1) + "/" + uom.getDate();
 70         return new Date(uom);
 71 	}
 72 
 73 	return calendar;
 74 
 75 },{requires: ["core","calendar"]});
