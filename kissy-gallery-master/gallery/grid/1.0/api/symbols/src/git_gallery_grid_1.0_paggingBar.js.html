1 KISSY.add("gallery/grid/1.0/paggingBar",function(S,bar){	
  2 
  3 	/** 
  4 		@exports S.LP as KISSY.LP
  5 	*/
  6 
  7 	var CLS_BTN_CUSTOM = "bar-btn-custom",//自定义按钮，不应用一般按钮的悬浮效果
  8 		CLS_INPUT_CONTAINER = 'pb-input-container',
  9 		CLS_IEXT_CONTAINER = 'pb-text-container',
 10 		CLS_TEXT_COUNT = 'pb-total-count',
 11 		CLS_TEXT_PAGE = 'pb-total-page',
 12 		CLS_FIRST = 'pb-page-first',
 13 		CLS_LAST = 'pb-page-last',
 14 		CLS_NEXT = 'pb-page-next',
 15 		CLS_PREV = 'pb-page-prev',
 16 		CLS_OK = 'pb-page-ok',
 17 		CLS_PAGGING_BAR = 'pagging-bar';
 18 	
 19 	//分页栏
 20 	/**
 21 	* 翻页时触发
 22 	* @name Grid.PaggingBar#beforepagechange
 23 	* @event  
 24 	* @param {event} e  事件对象
 25 	* @param {event} e.from  起始页
 26 	* @param {event} e.to  目标页
 27 	* @return {Boolean} 如果返回false则，取消翻页
 28 	*/
 29 
 30 	/**
 31 	* 分页控件
 32 	* @name Grid.PaggingBar
 33 	* @class
 34 	* @param {object} config 分页栏的配置项
 35 	* @param {Number} config.pageSize 单页数据量
 36 	* @param {Grid.Store} config.store 数据缓冲对象
 37 	* @example
 38 	* //应用于表格、列表时，会默认添加表格、列表的数据源，添加容器的编号
 39 	* tbar : {pageSize:20}
 40 	* //独立生成分页栏
 41 	* new Grid.PaggingBar({renderTo:'pbar',store:store,pageSize:20});
 42 	*/
 43 	function paggingBar(config){
 44 		var _self = this;
 45 		config = _self._formatConfig(config || {});
 46 		S.mix(_self, {
 47 			start : 0,  //页面的起始记录
 48             end : 0,    //页面的结束记录
 49             totalCount : 0, //记录的总数
 50             curPage : 1,    //当前页
 51             totalPage : 1  //总页数
 52 		});
 53 		paggingBar.superclass.constructor.call(_self, config);
 54 	}
 55 	
 56 	S.extend(paggingBar, bar);
 57 	S.augment(paggingBar,
 58 	/** @lends Grid.PaggingBar.prototype */	
 59 	{
 60 		CLS_BAR : CLS_PAGGING_BAR,
 61 		//数据加载完成后，更新状态
 62 		_afterStoreLoad : function (store, params) {
 63 			var _self = this,
 64 				pageSize = _self.get('pageSize'),
 65 				start = 0,  //页面的起始记录
 66 				end = 0,    //页面的结束记录
 67 				totalCount = 0, //记录的总数
 68 				curPage = 1,    //当前页
 69 				totalPage = 1;//总页数;
 70             if (params) {
 71                 start = params.start || 0;
 72             } else {
 73                 start  = 0;
 74             }
 75             //设置加载数据后翻页栏的状态
 76             totalCount = store.getTotalCount();
 77             end = totalCount - start > pageSize ? start + store.getCount() : totalCount;
 78             totalPage = parseInt((totalCount + pageSize - 1) / pageSize, 10);
 79             totalPage = totalPage > 0 ? totalPage : 1;
 80             curPage = parseInt(start / pageSize, 10) + 1;
 81 
 82 			_self.start = start;
 83 			_self.end = end;
 84 			_self.totalCount = totalCount;
 85 			_self.curPage = curPage;
 86 			_self.totalPage = totalPage;
 87 
 88             //设置按钮状态
 89             _self._setAllButtonsState();
 90             _self._setNumberPages();
 91         },
 92 		//分页栏的模版（容器）
 93 		_getBarTemplate : function() {
 94 			return '<div class="fr"></div>';
 95 		},
 96 		//获取按钮的配置
 97 		_getButtonConfig : function(id,buttonClass, disabled,text){
 98 			var _self = this;
 99 			return {
100 				id : id,
101 				type : 'button',
102 				css : buttonClass,
103 				text : text,
104 				containerCss : text ? CLS_BTN_CUSTOM : '',
105 				disabled : !!disabled,
106 				handler : function(event){
107 					var item = _self._findItem(event.target),
108 						btn = S.one('button',item);
109 					if(!DOM.hasClass(item,CLS_DISABLE)){
110 						_self._pageAction(btn);
111 					}
112 				}
113 			};
114 		},
115 		//获取分页配置
116 		_getPageInfoConfig : function(){
117 			var temp = ['<span class="', CLS_IEXT_CONTAINER, '">第</span><span class="', CLS_INPUT_CONTAINER, '"><input type="text" autocomplete="off" class="', CLS_ITEM_INPUT, '" size="20" name="inputItem"></span><span class="', CLS_IEXT_CONTAINER, ' ', CLS_TEXT_PAGE, '">页 共 0 页</span>'].join('');
118 			return {type:'custom',template : temp};
119 		},
120 		//获取总页数配置
121 		_getCountInfoConfig : function(){
122 			var temp = ['<div class="', CLS_IEXT_CONTAINER, '">共 <span class="', CLS_TEXT_COUNT, '">0</span> 条记录</div>'].join('');
123 			return {type:'custom',template : temp};
124 		},
125 		//格式化参数，提供分页栏的各个配置项
126 		_formatConfig : function (config){
127 			var _self = this,
128 				items = [],
129 				result =S.merge(config,{items:items});
130 			//保存 按钮（首页、最后一页、下一页、前一页）
131 			items.push(_self._getButtonConfig('btnfirst',CLS_FIRST,true));
132 			items.push(_self._getButtonConfig('btnpre',CLS_PREV,true));
133 
134 			//分割条
135 			items.push({type:'seperator'});
136 			//分页信息
137 			items.push(_self._getPageInfoConfig());
138 			
139 			items.push({type:'seperator'});
140 			items.push(_self._getButtonConfig('btnnext',CLS_NEXT,true));
141 			items.push(_self._getButtonConfig('btnlast',CLS_LAST,true));
142 			items.push(_self._getButtonConfig('btnok',CLS_OK,false,'跳转'));//确定按钮
143 
144 			items.push({type:'seperator'});
145             items.push(_self._getCountInfoConfig());//总计多少条
146 
147 			return result;
148 		},
149 		//初始化事件
150 		_initEvent : function (){
151 			//调用基类的初始化事件
152 			this.constructor.superclass._initEvent.call(this);
153 
154 			var _self = this,
155 				store = _self.get('store'),
156 				bar  = _self.get('bar');
157 
158 			if (store) {
159 				store.on('load', function (params) {
160 					_self._afterStoreLoad(store, params);
161 				});
162 			}
163 
164 			bar.one('.' + CLS_ITEM_INPUT).on('keyup', function (event) {
165 				event.stopPropagation();
166 				if (event.keyCode === 13) {
167 					var sender = S.one(this),
168 						enterValue = parseInt(sender.val(), 10);
169 					if (_self._isPageAllowRedirect(enterValue)) {
170 						_self.skipToPage(enterValue);
171 					} else {
172 						sender.val(_self.curPage);
173 					}
174 				}
175 			});
176 		},
177 		//是否允许跳转，若已经在当前页或者不在有效页范围内，不允许跳转
178 		_isPageAllowRedirect : function(value) {
179 			var _self = this;
180 			return value && value > 0 && value <= _self.totalPage && value !== _self.curPage;
181 		},
182 		//分页栏上的按钮事件
183 		_pageAction : function (btnEl) {
184 			var _self = this,
185 				page = 1;
186 			if (btnEl.hasClass(CLS_FIRST)) {
187 				page = 1;
188 			} else if (btnEl.hasClass(CLS_NEXT)) {
189 				page = _self.curPage + 1;
190 			} else if (btnEl.hasClass(CLS_PREV)) {
191 				page = _self.curPage - 1;
192 			} else if (btnEl.hasClass(CLS_LAST)) {
193 				page = _self.totalPage;
194 			} else if (btnEl.hasClass(CLS_OK)) {
195 				var bar = _self.get('bar');
196 					inputEl = bar.one('.' + CLS_ITEM_INPUT),
197 					value = parseInt(inputEl.val(), 10);
198 				if(_self._isPageAllowRedirect(value)){
199 					page = value;
200 				}else{
201 					inputEl.val(_self.curPage);
202 					return;
203 				}
204 			}
205 			_self.skipToPage(page);
206 
207 		},
208 		/**
209 		* 跳到指定页
210 		* @param {Number} page 目标页
211 		*/
212 		skipToPage : function (page) {
213 			var _self = this,
214 				store = _self.get('store'),
215 				pageSize = _self.get('pageSize'),
216 				index = page - 1,
217 				start = index * pageSize;
218 			var reslut = _self.fire('beforepagechange', {from : _self.curPage, to : page});
219 			if(store && reslut !== false){
220 				store.load({ start : start, limit : pageSize, pageIndex : index });
221 			}
222 		},
223 		//设置按钮状态
224 		_setAllButtonsState : function () {
225 			var _self = this,
226 				store = _self.get('store'),
227 				items = _self.get('items');
228 			
229 			/**
230 			* @private
231 			*/
232 			function enableButtons(btnNames, enable) {
233 				S.each(items, function (item) {
234 					if(S.inArray(item.id,btnNames)){
235 						item.set('disabled',!enable);
236 					}
237 				});
238 			}
239 			if (store) {
240                 enableButtons(['btnpre', 'btnnext', 'btnfirst', 'btnlast', 'btnok'], true);
241             }
242 
243             if (_self.curPage === 1) {
244                 enableButtons(['btnpre', 'btnfirst'], false);
245             }
246             if (_self.curPage === _self.totalPage) {
247                 enableButtons(['btnnext', 'btnlast'], false);
248             }
249 		},
250 		//显示页码和总页数
251 		_setNumberPages : function () {
252 			var _self = this,
253 				bar = _self.get('bar'),
254 				input = bar.one('.' + CLS_ITEM_INPUT),
255 				text = bar.one('.' + CLS_TEXT_PAGE),
256 				totalEl = bar.one('.' + CLS_TEXT_COUNT);
257 			if (input && text) {
258 				input.val(_self.curPage);
259 				text.text('页 共 ' + _self.totalPage + ' 页');
260 			}
261 			if (totalEl) {
262 				totalEl.text(_self.totalCount);
263 			}
264 		}
265 	});
266 	
267 	//S.namespace('LP');
268 
269 	return paggingBar;
270 }, {requires : ["./bar"]});
