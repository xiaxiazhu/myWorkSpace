  2 KISSY.add("gallery/grid/1.0/bar",function(S,barItem){
  3 	var DOM = S.DOM,Node = S.Node;
  4 
  5 	var CLS_BTN_CONTAINER = 'bar-btn-container';
  6 
  7 	//分页栏、工具栏、状态栏的基类
  8 	function bar(config){
  9 		var _self = this;
 10 		config = config || {};
 11 		
 12 		bar.superclass.constructor.call(_self, config);
 13 		_self._init();
 14 	}
 15 
 16 	
 17 	S.extend(bar, S.Base);
 18 	S.augment(bar,{
 19 		//Bar 的自定义样式
 20 		CLS_BAR : '',
 21 		//查找 子元素的DOM
 22 		_findItem : function (element) {
 23 			if (DOM.hasClass(element, CLS_BTN_CONTAINER)) {
 24 				return element;
 25 			} else {
 26 				return DOM.parent(element, '.' + CLS_BTN_CONTAINER);
 27 			}
 28 		},
 29 		//提供的默认模板
 30 		_getBarTemplate : function(){
 31 			return '<div></div>';
 32 		},
 33 		//初始化配置项
 34 		_init : function(){
 35 			var _self = this,
 36 				items = _self.get('items'),
 37 				defaults = _self.get('defaults'),
 38 				item = null,//配置项
 39 				cls = null; //配置项类型
 40 			for(var i = 0,count = items.length;i < count;i++){
 41 				item = S.merge(defaults, items[i]);
 42 				if(S.isUndefined(item.id)){
 43 					item.id = i;
 44 				}
 45 				if(!item.isBarItem){
 46 					cls = barItem.types[item.type||'text'];
 47 					if(!cls){
 48 						cls = barItem.types['text'];
 49 					}
 50 					items[i] = new cls(item);
 51 				}
 52 			}
 53 			_self._initDOM();
 54 			_self._initEvent();
 55 		},
 56 		//初始化Dom元素
 57 		_initDOM : function(){
 58 			var _self = this,
 59 				barTemplate = _self._getBarTemplate(),
 60 				renderTo = _self.get('renderTo'),
 61 				container = DOM.get('#'+renderTo),
 62 				items = _self.get('items'),
 63 				barEl = new Node(barTemplate).appendTo(container),
 64 				tableEl = new Node('<table style="border-spacing: 0;"><tr></tr></table>').appendTo(barEl),
 65 				tr = DOM.get('tr',tableEl[0]);
 66 			_self.set('bar', barEl);
 67 			
 68 			//创建bar各个子项的Dom
 69 			S.each(items,function (item,index){
 70 				var cell = tr.insertCell(index);
 71 				item.renderTo(cell);
 72 			});
 73 			DOM.addClass(container,_self.CLS_BAR);
 74 			DOM.show(container);
 75 		},
 76 		//初始化各个子项的事件
 77 		_initEvent : function(){
 78 			var items = this.get('items');
 79 			S.each(items,function (item){
 80 				item.attachEvent();
 81 			});
 82 		}
 83 	});
 84 	return bar;
 85 
 86 }, {requires : ["./barItem"]});
