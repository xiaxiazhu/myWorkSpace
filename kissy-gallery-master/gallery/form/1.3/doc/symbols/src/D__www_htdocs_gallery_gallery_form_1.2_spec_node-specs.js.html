1 var KISSY = require('KISSY');
  2 var jasmine = require('jasmine-node');
  3 var path = require('path');
  4 
  5 KISSY.config({
  6     packages: [
  7         {
  8             name: 'form',
  9             path: path.resolve('.')+'/src',
 10             charset: 'gbk'
 11         }
 12     ]
 13 });
 14 
 15 for (var key in jasmine) {
 16   global[key] = jasmine[key];
 17 }
 18 
 19 global.prepare = function(mods, fn) {
 20     beforeEach(function() {
 21         var ok = 0;
 22         KISSY.use(mods, function() {
 23             ok = 1;
 24             var that = this, args = [].slice.call(arguments);
 25             runs(function() {
 26                 fn.apply(that, args);
 27             });
 28         });
 29         waitsFor(function() {
 30             return ok;
 31         }, mods + ' is not loaded.', 10000);
 32     });
 33 };
 34 
 35 var isVerbose = false;
 36 var showColors = true;
 37 process.argv.forEach(function(arg) {
 38   switch (arg) {
 39   case '--color': showColors = true; break;
 40   case '--noColor': showColors = false; break;
 41   case '--verbose': isVerbose = true; break;
 42   }
 43 });
 44 
 45 var getTestFile = function() {
 46     var c = process.argv[2];
 47     return path.resolve(__dirname, c);
 48 };
 49 
 50 jasmine.executeSpecsInFolder(getTestFile(), function(runner, log) {
 51     console.log('Running on: ' + __dirname + getTestFile());
 52     process.exit(runner.results().failedCount == 0 ? 0 : 1);
 53 }, isVerbose, showColors, undefined, undefined, /spec\.js$/, {});
