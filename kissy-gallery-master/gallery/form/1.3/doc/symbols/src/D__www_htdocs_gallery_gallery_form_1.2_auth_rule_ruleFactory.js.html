1 /**
  2  * @fileoverview html 属性规则工厂
  3  * @author 张挺 <zhangting@taobao.com>
  4  *
  5  */
  6 KISSY.add('gallery/form/1.3/auth/rule/ruleFactory', function (S, Base, PropertyRule, Rule, undefined) {
  7     var RuleFactory = function () {
  8         var self = this;
  9 
 10         RuleFactory.superclass.constructor.call(self);
 11     };
 12 
 13     RuleFactory.rules = {};
 14 
 15     //第一个参数一定是属性的value，后面的才是真正的参数
 16     S.mix(RuleFactory.rules, {
 17         required:function (pv, value) {
 18             if(S.isArray(value)) {
 19                 return value.length>0;
 20             }
 21 
 22             return !!value;
 23         },
 24         pattern:function (pv, value) {
 25             return new RegExp(pv).test(value);
 26         },
 27         max:function (pv, value) {
 28             if (!S.isNumber(value)) {
 29                 return false;
 30             }
 31             return value <= pv;
 32         },
 33         min:function (pv, value) {
 34             if (!S.isNumber(value)) {
 35                 return false;
 36             }
 37             return value >= pv;
 38         },
 39         step:function (pv, value) {
 40             if (!S.isNumber(value)) {
 41                 return false;
 42             }
 43             if(value == 0 || pv == 1) return true;
 44 
 45             return value % pv;
 46         },
 47         //添加1个特殊的属性
 48         equalTo:function(pv, value){
 49             //number same
 50             if (S.isNumber(value)) {
 51                 return pv === value;
 52             }
 53 
 54             //selector same
 55             if(S.one(pv)) {
 56                 return S.one(pv).val() === value;
 57             }
 58 
 59             //string same
 60             return pv === value;
 61         }
 62     });
 63 
 64     S.mix(RuleFactory, {
 65         HTML_PROPERTY:['required', 'pattern', 'max', 'min', 'step', 'equalTo'],
 66         register:function(name, rule) {
 67             RuleFactory.rules[name] = rule;
 68         },
 69         create:function (ruleName, cfg) {
 70             if(S.inArray(ruleName, RuleFactory.HTML_PROPERTY)) {
 71                 return new PropertyRule(ruleName, RuleFactory.rules[ruleName], cfg);
 72             } else if(RuleFactory.rules[ruleName]) {
 73                 return new Rule(ruleName, RuleFactory.rules[ruleName], cfg);
 74             }
 75             return undefined;
 76         }
 77     });
 78 
 79     return RuleFactory;
 80 
 81 }, {
 82     requires:[
 83         'base',
 84         './html/propertyRule',
 85         './rule'
 86     ]
 87 });
