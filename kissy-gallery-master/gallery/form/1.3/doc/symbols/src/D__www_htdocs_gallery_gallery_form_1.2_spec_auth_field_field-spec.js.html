  6 describe('field test suite', function() {
  7     KISSY.use('form/validation/field/field, dom, json', function(S, Field, D, JSON) {
  8         var $ = S.all;
  9         beforeEach(function() {
 10             D.remove('#J_Test');
 11         });
 12 
 13         it('create html field', function() {
 14             $('body').append('<input value="3" required  id="J_Test"');
 15             var f = new Field('#J_Test');
 16             expect(f).toBeDefined();
 17         });
 18 
 19         it('more property', function() {
 20             $('body').append('<input value="x" required  pattern="[0-9]" id="J_Test"');
 21             var f = new Field('#J_Test');
 22 
 23             //validate all
 24             expect(f.validate()).toBeFalsy();
 25 
 26             //validate single rule
 27             expect(f.validate('required')).toBeTruthy();
 28 
 29             //validate pattern
 30             expect(f.validate('pattern')).toBeFalsy();
 31 
 32             //use new value to validation
 33             $('#J_Test').val(2);
 34             expect(f.validate('pattern')).toBeTruthy();
 35         });
 36 
 37         xit('use property config json', function() {
 38             $('body').append('<input value="x" required  pattern="[0-9]" id="J_Test"');
 39             var valid = {
 40                 required:'hello world',
 41                 pattern:'good pattern'
 42             };
 43 
 44             S.one('#J_Test').attr('data-valid', JSON.parse(valid));
 45             var f = new Field('#J_Test');
 46 
 47             //validate all
 48             expect(f.validate()).toBeFalsy();
 49 
 50             //validate single rule
 51             expect(f.validate('required')).toBeTruthy();
 52         });
 53 
 54         it('use json param', function() {
 55             $('body').append('<input value="x" required pattern="[0-9]" id="J_Test"');
 56             var valid = {
 57                 event:'focus',
 58                 rules:{
 59                     required: {
 60                         success:'hello world',
 61                         error:'required error'
 62                     },
 63                     pattern:{
 64                         success:'good pattern',
 65                         error:'pattern error'
 66                     }
 67                 }
 68             };
 69 
 70             var f = new Field('#J_Test', valid);
 71             $('#J_Test').fire('blur');
 72             expect(f.get('message')).toEqual('');
 73 
 74             $('#J_Test').fire('focus');
 75             expect(f.get('message')).toEqual('pattern error');
 76 
 77             //change value
 78             $('#J_Test').val('2');
 79             f.on('validate', function(e){
 80                 expect(e.msg).toEqual('good pattern');
 81             });
 82             f.validate();
 83             expect(f.get('message')).toEqual('good pattern');
 84         });
 85 
 86         it('use simple message json param2', function() {
 87             $('body').append('<input value="x" required pattern="[0-9]" id="J_Test"');
 88             var valid = {
 89                 event:'focus',
 90                 rules:{
 91                     required:'required error',
 92                     pattern:'pattern error'
 93                 }
 94             };
 95 
 96             var f = new Field('#J_Test', valid);
 97 
 98             $('#J_Test').fire('focus');
 99             expect(f.get('message')).toEqual('pattern error');
100         });
101 
102         it('get success rule msg', function() {
103             $('body').append('<input value="1" required pattern="[0-9]" id="J_Test"');
104             var valid = {
105                 event:'focus',
106                 rules:{
107                     required: {
108                         success:'hello world',
109                         error:'required error'
110                     },
111                     pattern:{
112                         success:'good pattern',
113                         error:'pattern error'
114                     }
115                 }
116             };
117 
118             var f = new Field('#J_Test', valid);
119 
120             $('#J_Test').fire('focus');
121             expect(f.get('message')).toEqual('good pattern');
122         });
123 
124         xit('test equalto property', function() {
125             $('body').append('<input value="1" equalTo="#J_Test1" id="J_Test"');
126             $('body').append('<input value="2"  id="J_Test1"');
127             var valid = {
128                 event:'focus',
129                 rules:{
130                     equalTo: {
131                         success:'the same',
132                         error:'not the same'
133                     }
134                 }
135             };
136 
137             var f = new Field('#J_Test', valid);
138 
139             f.validate();
140             expect(f.get('message')).toEqual('not the same');
141         });
142 
143         it('add custom rule', function() {
144             $('body').append('<input value="4" required  pattern="[0-9]" id="J_Test"');
145             var f = new Field('#J_Test');
146             f.add('test', function(value) {
147                 return value == 1;
148             }, {
149                 msg:'test fail'
150             });
151 
152             //����ʼ��û�з���value����ô�Զ����?��value����
153             expect(f.validate()).toBeFalsy();
154         });
155 
156         xit('group validation', function() {
157             $('body').append('<input value="x" required pattern="[0-9]" id="J_Test"');
158             var valid = {
159                 required:'hello world',
160                 pattern:'good pattern',
161                 groups:{
162                     a:['require']
163                 }
164             };
165 
166             S.one('#J_Test').attr('data-valid', JSON.parse(valid));
167             var f = new Field('#J_Test');
168 
169             //validate group
170             expect(f.validate({
171                 groups:['a']
172             })).toBeTruthy();
173         });
174     });
