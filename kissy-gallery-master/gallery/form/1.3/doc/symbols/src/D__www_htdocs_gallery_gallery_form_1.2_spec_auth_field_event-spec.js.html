 10 KISSY.use("event", function(S, Event) {
 11 
 12 
 13     describe("custom event is same as native event.", function() {
 14 
 15         it("can fire more than one", function() {
 16 
 17             var args = [];
 18 
 19             function Test() {
 20                 this.on("test", function(e) {
 21                     args.push(e.a);
 22                 });
 23                 this.on("test2", function(e) {
 24                     args.push(e.a);
 25                 });
 26             }
 27 
 28             S.augment(Test, Event.Target);
 29 
 30             var t = new Test();
 31 
 32             t.fire("test test2", {
 33                 a:1
 34             });
 35 
 36             expect(args).toEqual([1,1]);
 37 
 38         });
 39 
 40 
 41         it("can bubble", function() {
 42 
 43             var ret = [],args = [];
 44 
 45             function Test() {
 46                 this.publish("test", {
 47                     bubbles:1
 48                 });
 49 
 50                 this.on("test", function(e) {
 51                     ret.push(this.id);
 52                     args.push(e.a);
 53                     e.a++;
 54                 });
 55             }
 56 
 57             S.augment(Test, Event.Target);
 58 
 59             var t = new Test();
 60 
 61             t.id = 1;
 62 
 63 
 64             var t2 = new Test();
 65 
 66             t2.id = 2;
 67 
 68             t2.addTarget(t);
 69 
 70             t2.fire("test", {
 71                 a:1
 72             });
 73 
 74             expect(ret).toEqual([2,1]);
 75             expect(args).toEqual([1,2]);
 76 
 77             ret = [];
 78             args = [];
 79 
 80             t2.removeTarget(t);
 81 
 82             t2.fire("test", {
 83                 a:1
 84             });
 85 
 86             expect(ret).toEqual([2]);
 87             expect(args).toEqual([1]);
 88 
 89         });
 90 
 91 
 92         it("can stop bubble by stopPropagation()", function() {
 93             var ret = [];
 94 
 95             function Test() {
 96                 this.publish("test", {
 97                     bubbles:1
 98                 });
 99 
100                 this.on("test", function(e) {
101                     ret.push(this.id);
102                     e.stopPropagation();
103                 });
104             }
105 
106             S.augment(Test, Event.Target);
107 
108             var t = new Test();
109 
110             t.id = 1;
111 
112 
113             var t2 = new Test();
114 
115             t2.id = 2;
116 
117             t2.addTarget(t);
118 
119             t2.fire("test");
120 
121             expect(ret).toEqual([2]);
122         });
123     });
124 
125 });
