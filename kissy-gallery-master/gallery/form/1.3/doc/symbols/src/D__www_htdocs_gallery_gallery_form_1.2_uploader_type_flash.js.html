1 /**
  2  * @fileoverview flash上传方案，基于龙藏写的ajbridge内的uploader
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.3/uploader/type/flash', function (S, Node, UploadType, swfUploader) {
  6     var EMPTY = '', $ = Node.all, LOG_PREFIX = '[uploader-FlashType]:';
  7 
  8     /**
  9      * @name FlashType
 10      * @class flash上传方案，基于龙藏写的ajbridge内的uploader
 11      * @constructor
 12      * @extends UploadType
 13      * @requires Node
 14      */
 15     function FlashType(config) {
 16         var self = this;
 17         //调用父类构造函数
 18         FlashType.superclass.constructor.call(self, config);
 19         self.isHasCrossdomain();
 20         self._init();
 21     }
 22 
 23     S.mix(FlashType, /** @lends FlashType.prototype*/{
 24         /**
 25          * 事件列表
 26          */
 27         event:S.merge(UploadType.event, {
 28             //swf文件已经准备就绪
 29             SWF_READY: 'swfReady',
 30             //正在上传
 31             PROGRESS:'progress'
 32         })
 33     });
 34 
 35     S.extend(FlashType, UploadType, /** @lends FlashType.prototype*/{
 36         /**
 37          * 初始化
 38          */
 39         _init:function () {
 40             var self = this, swfUploader = self.get('swfUploader');
 41             if(!swfUploader){
 42                 S.log(LOG_PREFIX + 'swfUploader对象为空！');
 43                 return false;
 44             }
 45             //SWF 内容准备就绪
 46             swfUploader.on('contentReady', function(ev){
 47                 self.fire(FlashType.event.SWF_READY);
 48             }, self);
 49             //监听开始上传事件
 50             swfUploader.on('uploadStart', self._uploadStartHandler, self);
 51             //监听文件正在上传事件
 52             swfUploader.on('uploadProgress', self._uploadProgressHandler, self);
 53             //监听文件上传完成事件
 54             swfUploader.on('uploadCompleteData',self._uploadCompleteDataHandler,self);
 55             //监听文件失败事件
 56             swfUploader.on('uploadError',self._uploadErrorHandler,self);
 57         },
 58         /**
 59          * 上传文件
 60          * @param {String} id 文件id
 61          * @return {FlashType}
 62          */
 63         upload:function (id) {
 64             var self = this, swfUploader = self.get('swfUploader'),
 65                 action = self.get('action'), method = 'POST',
 66                 data = self.get('data'),
 67                 name = self.get('fileDataName');
 68             if(!name) name = 'Filedata';
 69             self.set('uploadingId',id);
 70             S.mix(data,{"type":"flash"});
 71             swfUploader.upload(id, action, method, data,name);
 72             return self;
 73         },
 74         /**
 75          * 停止上传文件
 76          * @return {FlashType}
 77          */
 78         stop:function () {
 79             var self = this, swfUploader = self.get('swfUploader'),
 80                 uploadingId = self.get('uploadingId');
 81             if(uploadingId != EMPTY){
 82                 swfUploader.cancel(uploadingId);
 83                 self.fire(FlashType.event.STOP, {id : uploadingId});
 84             }
 85             return self;
 86         },
 87         /**
 88          * 开始上传事件监听器
 89          * @param {Object} ev ev.file：文件数据
 90          */
 91         _uploadStartHandler : function(ev){
 92             var self = this;
 93             self.fire(FlashType.event.START, {'file' : ev.file });
 94         },
 95         /**
 96          * 上传中事件监听器
 97          * @param {Object} ev
 98          */
 99         _uploadProgressHandler:function (ev) {
100             var self = this;
101             S.mix(ev, {
102                 //已经读取的文件字节数
103                 loaded:ev.bytesLoaded,
104                 //文件总共字节数
105                 total : ev.bytesTotal
106             });
107             S.log(LOG_PREFIX + '已经上传字节数为：' + ev.bytesLoaded);
108             self.fire(FlashType.event.PROGRESS, { 'loaded':ev.loaded, 'total':ev.total });
109         },
110         /**
111          * 上传完成后事件监听器
112          * @param {Object} ev
113          */
114         _uploadCompleteDataHandler : function(ev){
115             var self = this,result;
116             try {
117                 result = JSON.parse(ev.data);
118             } catch(err) {
119                 S.log(LOG_PREFIX + 'json数据格式不合法！');
120                 self.fire(FlashType.event.ERROR, {msg : '不是合法的json数据'});
121             }
122             S.log(LOG_PREFIX + '服务器端输出：' + result);
123             self.set('uploadingId',EMPTY);
124             self.fire(FlashType.event.SUCCESS, {result : result});
125         },
126         /**
127          *文件上传失败后事件监听器
128          */
129         _uploadErrorHandler : function(ev){
130             var self = this;
131             self.set('uploadingId',EMPTY);
132             self.fire(FlashType.event.ERROR, {msg : ev.msg});
133         },
134         /**
135          * 应用是否有flash跨域策略文件
136          */
137         isHasCrossdomain:function(){
138             var domain = location.hostname;
139              S.io({
140                  url:'http://' + domain + '/crossdomain.xml',
141                  dataType:"xml",
142                  error:function(){
143                      S.log('缺少crossdomain.xml文件或该文件不合法！');
144                  }
145              })
146         }
147     }, {ATTRS:/** @lends FlashType*/{
148         /**
149          * 服务器端路径，留意flash必须是绝对路径
150          */
151         action:{
152             value:EMPTY,
153             getter:function(v){
154                 var reg = /^http/;
155                 //不是绝对路径拼接成绝对路径
156                 if(!reg.test(v)){
157                      var href = location.href,uris = href.split('/'),newUris;
158                     newUris  = S.filter(uris,function(item,i){
159                         return i < uris.length - 1;
160                     });
161                     v = newUris.join('/') + '/' + v;
162                 }
163                 return v;
164             }
165         },
166         /**
167          * ajbridge的uploader组件的实例，必须参数
168          */
169         swfUploader:{value:EMPTY},
170         /**
171          * 正在上传的文件id
172          */
173         uploadingId : {value : EMPTY}
174     }});
175     return FlashType;
176 }, {requires:['node', './base']});
