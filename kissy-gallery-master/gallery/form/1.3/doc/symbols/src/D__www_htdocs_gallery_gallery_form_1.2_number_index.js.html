1 /**
  2  * @fileoverview 数字文本框
  3  * @author 易敛<yilian.wj@taobao.com>
  4  * @date 12-4-22
  5  */
  6 KISSY.add('gallery/form/1.3/number/index', function(S, Node, Base){
  7 	var $ = Node.all, D = S.DOM;
  8 	/**
  9 	 * @name Number
 10 	 * @class 数字文本框
 11 	 * @constructor
 12 	 * @extends Base
 13 	 * @param {String} target 目标
 14 	 * @param {Object} config 组件配置
 15 	 * @example
 16 	 * var ck = new Number('#J_Content input',{trigger:{plus:'#J_Plus',minus:'#J_Minus'}})
 17 	 */
 18 	function Number(target, config) {
 19 		
 20 		var self = this;
 21 		config = S.merge({target: $(target)},config);
 22 		//调用父类构造器
 23 		Number.superclass.constructor.call(self, config);
 24 	}
 25 
 26 	S.extend(Number, Base,  /** @lends Number.prototype*/{
 27 		/*
 28 		* 运行
 29 		*/
 30 		render: function(){
 31 			var self = this,$target = self.get('target');
 32             if(!$target.length) return false;
 33             self.addHTML();
 34             self.eventOnChangeNum();
 35             self.eventOnValide();
 36 		},
 37 		/**
 38 		 * 对input组件进行包装
 39 		 */
 40 		addHTML: function(){
 41 			var self = this, $target = self.get('target'), getCls = this.get('cls'), accessible = self.get('accessible');
 42 			//创建元素
 43 			var $parent = $target.parent(),
 44 			$containerEl = $(D.create('<span class="'+ getCls.container + '">')),
 45 			$plusEl = $(D.create('<a>',{href: '#!/price/plus', text: '-'})),
 46 			$minusEl = $(D.create('<a>',{href: '#!/price/minus', text: '+'}));
 47 			$plusEl.addClass(getCls.plus);
 48 			$minusEl.addClass(getCls.minus);
 49 
 50 			if(accessible){
 51 				$target.addAttr('aria-label','出价框，请输入价格');
 52 			}
 53 
 54 			//建立元素之间关系
 55 			$containerEl.append($minusEl).append($target).append($plusEl);
 56 			$parent.append($containerEl);
 57 
 58 		},
 59 
 60 		eventOnChangeNum: function(){
 61 			var self = this, $target = self.get('target'), inputValue, range = +$target.attr('data-range') || 1,
 62 			getTrigger = self.get('trigger'), $plus = $(getTrigger.plus), $minus = $(getTrigger.minus),
 63 			limitRange = self.limitRange;
 64 			$plus.on('click', function(){	
 65 				inputValue = +$target.val();
 66 				inputValue += range;
 67 				limitRange.call(self,  inputValue);
 68 			});
 69 			$minus.on('click',function(){
 70 				
 71 				inputValue = +$target.val();
 72 				inputValue -= range;
 73 				limitRange.call(self, inputValue);
 74 			})
 75 		},
 76 		eventOnValide: function(){
 77 			var self = this, $target = self.get('target'),formatNum = self.formatNum, limitRange = self.limitRange;
 78 			$target.on('blur',function(){
 79 				formatNum.call(self, $target);
 80 				limitRange.call(self, +$target.val());
 81 			})
 82 		},
 83 
 84 		toFloat: function(value){
 85             return parseFloat(value, 10);			
 86 		},
 87 
 88         /**
 89 		 * [numValidation 校正输入框格式，屏蔽非法字符]
 90 		 * @param  {[HTMLELement]} $target [文本框节点]
 91 		 */
 92         formatNum : function($target){
 93     		var self = this, min = +$target.attr('data-min') || 1,
 94     		inputValue = self.toFloat($target.val().replace(/[^\d\.]/g, ''));
 95     		inputValue = isNaN(inputValue) ? min : Math.max(inputValue, min);
 96     		$target.val(inputValue.toFixed(2));
 97         },
 98         /**
 99          * [limitRange 控制输入数值的大小在最大值与最小值之间]
100          * @param  {[Number]} value [输入框的值]
101          */
102         limitRange : function(value){
103         	var self = this, $target = self.get('target'), toFloat = self.toFloat,
104         	min = toFloat($target.attr('data-min')) || 1,
105 			max = toFloat($target.attr('data-max')) || 1,
106         	inputValue = min && Math.max(min, value);
107 			inputValue = max && Math.min(max, inputValue);
108 			$target.val(inputValue.toFixed(2));
109         }
110 	},{
111 		ATTRS: /** @lends Number.prototype*/{
112 			/**
113 			 * 配置的目标,选择器的字符串
114 			 * @type {String}
115 			 */
116 			target: {
117 				value: '',
118 				setter: function(v) {
119 					return $(v);
120 				},
121 				getter: function(v) {
122 					return $(v);
123 				}
124 			},
125 
126 			/*
127 			* 触发加减的按钮
128 			*/
129 			trigger: {
130 				value: {
131 					plus: '.ks-number-plus',
132 					minus: '.ks-number-minus'
133 				}
134 			},
135 			/**
136              * 一组样式名
137              * @type {Object}
138              * @default cls:{init: 'ks-radio',selected: 'ks-radio-selected',disabled: 'ks-radio-disabled',hover: 'ks-radio-hover'}
139              */
140             cls: {
141                 value: {
142                     init: 'ks-price-input',
143                     plus: 'ks-number-plus',
144                     minus: 'ks-number-minus',
145                     container: 'ks-plus-minus-operation'
146                 }
147             },
148             /**
149              * 无障碍，设置aria属性
150              * @default false
151              */
152             accessible: {
153                 value: false
154             }
155 		}
156 	});
157 	return Number;
158 },{requires:['node','base','./index.css']});
