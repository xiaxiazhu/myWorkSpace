  6 KISSY.add('gallery/form/1.2/auth/msg/base', function (S, Base) {
  7 
  8     /**
  9      * msg cls
 10      * @type {String}
 11      */
 12     var AUTH_MSG_CLS = 'kf-msg';
 13 
 14     var Msg = function (srcNode, cfg) {
 15         var self = this;
 16 
 17         self._init(srcNode, cfg);
 18 
 19         Msg.superclass.constructor.call(self);
 20     };
 21 
 22 
 23     S.extend(Msg, Base, {
 24         /**
 25          * init msg
 26          * @param srcNode {htmlElement|String}
 27          * @param cfg {Object}
 28          * @private
 29          */
 30         _init:function (srcNode, cfg) {
 31             var self = this;
 32             self._el = S.one(srcNode);
 33             self.set('tpl', cfg.tpl);
 34             self.set('args', cfg.args);
 35 
 36             self._msgContainer = S.one('.' + AUTH_MSG_CLS, self._el.parent());
 37 
 38             if(!self._msgContainer) {
 39                 self._msgContainer = S.one('<div class="' + AUTH_MSG_CLS +'" style="display: none"></div>');
 40                 self._el.parent().append(self._msgContainer);
 41             }
 42 
 43         },
 44         hide:function () {
 45             this._msgContainer.hide();
 46         },
 47         show:function (o) {
 48             var self = this;
 49             o = S.merge(self.get('args'), o);
 50 
 51             S.buffer(function () {
 52                 self._msgContainer.html(S.substitute(self.get('tpl'), o));
 53                 self._msgContainer.show();
 54             }, 50)();
 55         }
 56     }, {
 57         ATTRS:{
 58             tpl:{
 59                 value:''
 60             },
 61             args:{
 62                 value:{}
 63             }
 64         }
 65     });
 66 
 67     return Msg;
 68 
 69 }, {
 70     requires:[
 71         'base'
 72     ]
 73 });
