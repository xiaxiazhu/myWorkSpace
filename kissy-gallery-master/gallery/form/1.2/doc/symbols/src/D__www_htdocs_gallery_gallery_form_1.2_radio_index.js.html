1 /**
  2  * @fileoverview  radio
  3  * @author 伯方<bofang.zxj@taobao.com>
  4  **/
  5 KISSY.add('gallery/form/1.2/radio/index', function(S, Node, Base) {
  6     var  $ = Node.all;
  7     /**
  8      * @name Radio
  9      * @class Radio美化组件
 10      * @constructor
 11      * @extends Base
 12      * @param {String} target 目标元素
 13      * @param {Object} config 配置
 14      * @example
 15      * var r = new Radio('#J_Radio input');
 16      * r.render();
 17      */
 18     function Radio(target, config) {
 19         var self = this;
 20         config = S.merge({
 21             target: target
 22         }, config);
 23         //调用父类构造函数
 24         Radio.superclass.constructor.call(self, config);
 25     }
 26     //方法
 27     S.extend(Radio, Base, /** @lends Radio.prototype*/ {
 28         /**
 29          * 开始执行
 30          */
 31         render: function() {
 32             var self = this;
 33             //加载css
 34             self._loadCss();
 35             //开始替换
 36             self._replaceRadio();
 37             //事件绑定
 38             self._bindEvent();
 39         },
 40         /**
 41          * 还原radio为原生的radio
 42          * @return {Object} return self
 43          */
 44         recoverRadio: function() {
 45             var self = this,
 46                 targets = self.get('target'),
 47                 radio = self.get('radio');
 48             $(radio).each(function(value, key) {
 49                 value.hide();
 50                 $(targets[key]).show();
 51             })
 52             //self.set('radio',[]);// = null;
 53             return self;
 54         },
 55         /**
 56          * 用span替换radio，关键步骤
 57          */
 58         _replaceRadio: function() {
 59             var self = this,
 60                 target = self.get('target'),
 61                 html = self._getHtml(0),
 62                 disabledHTML = self._getHtml(2),
 63                 selectedHTML = self._getHtml(1),
 64                 radio, radioArr = [],
 65                 accessible = self.get('accessible'),
 66                 getLabelFunc = self.get('getLabelFunc'),
 67                 labelText;
 68             if (target.length === 0) {
 69                 return false;
 70             }
 71             //遍历
 72             target.each(function(value, key) {
 73                 value.hide();
 74                 if (self._isDisabled(value)) {
 75                     radio = $(disabledHTML).insertBefore(value).attr('ks-radio-disabled', 'disabled').removeAttr('tabindex');
 76                 } else {
 77                     // 如果本身是选中的状态
 78                     radio = self._isSelected(value) ? $(selectedHTML) : $(html);
 79                     radio.insertBefore(value);
 80                 }
 81                 // 无障碍
 82                 if (accessible) {
 83                     try {
 84                         //优先选择函数提供的查询
 85                         labelText = getLabelFunc ? getLabelFunc(value).html() : value.next('label').html();
 86                         radio.attr('aria-label', labelText);
 87                     } catch (e) {
 88                         S.log('html结构不符合');
 89                         return false;
 90                     }
 91 
 92                 }
 93                 radioArr.push(radio);
 94             })
 95             self.set('radio', radioArr);
 96         },
 97         /**
 98          * 加载css
 99          */
100         _loadCss: function() {
101             var self = this,
102                 cssUrl = self.get('cssUrl');
103             //加载css文件
104             if (cssUrl !== '') {
105                 S.use(cssUrl, function(S) {});
106             }
107         },
108         /**
109          * 根据样式返回html字符串
110          * @param  {Number} key 0→DEFAULT;1→selected;2→DISABLED
111          * @return {String} 返回html
112          */
113         _getHtml: function(key) {
114             var self = this,
115                 getClass = self.get('cls'),
116                 defaultClass = getClass.init,
117                 selectedClass = getClass.selected,
118                 disabledClass = getClass.disabled,
119                 htmlStr = '<span tabindex="0" class="{defalutName} {secondName}"></span>',
120                 obj = {
121                     defalutName: defaultClass
122                 };
123             switch (key) {
124             case 0:
125                 obj.secondName = '';
126                 break;
127             case 1:
128                 obj.secondName = selectedClass;
129                 break;
130             case 2:
131                 obj.secondName = disabledClass;
132                 break;
133             default:
134                 break;
135             }
136             return S.substitute(htmlStr, obj);
137         },
138         /**
139          * 绑定事件，包括mouseenter mouseleave click
140          */
141         _bindEvent: function() {
142             var self = this,
143                 radio = $(self.get('radio')),
144                 hoverClass = this.get('cls').hover,
145                 hasLabel = self.get('hasLabel'),
146                 targets = self.get('target'),
147                 getLabelFunc = self.get('getLabelFunc'),
148                 nextLabel;
149             radio.each(function(value, key) {
150                 value.on('mouseenter mouseleave', function(ev) {
151                     //如果本身是选中状态或者是禁用状态，则不做处理
152                     if (self._isSelected(value) || self._isDisabled(value)) {
153                         return;
154                     }
155                     //value.toggleClass('ks-radio-hover') 在初始化的时候就已经选中的无效
156                     switch (ev.type) {
157                     case 'mouseenter':
158                         value.addClass(hoverClass);
159                         break;
160                     case 'mouseleave':
161                         value.removeClass(hoverClass);
162                         break;
163                     default:
164                         break;
165                     }
166                     //单击                
167                 }).on('click', function() {
168                     if (self._isDisabled(value)) return;
169                     self._clickHandler.call(self, key);
170                     //按键 enter
171                 }).on('keyup', function(ev) {
172                     if (ev.keyCode === 13) {
173                         value.fire('click');
174                     }
175                 });
176                 //如果需要 label-for
177                 if (hasLabel) {
178                     try {
179                         nextLabel = getLabelFunc ? getLabelFunc($(targets[key])) : value.next('label');
180                         //将label绑定和radio一样的事件
181                         nextLabel.on('click', function() {
182                             value.fire('click');
183                         }).on('mouseenter', function() {
184                             value.fire('mouseenter');
185                         }).on('mouseleave', function() {
186                             value.fire('mouseleave');
187                         })
188                     } catch (e) {
189                         S.log('html结构不符合');
190                         return false;
191                     }
192                 }
193             })
194         },
195         /**
196          * 单击事件
197          * @param  {Number} targetIndex 数组radio的索引
198          */
199         _clickHandler: function(targetIndex) {
200             var that = this,
201                 targets = that.get('target'),
202                 radios = $(that.get('radio'));
203             radio = $(that.get('radio')[targetIndex]), getCls = this.get('cls'), selectedClass = getCls.selected, hoverClass = getCls.hover;
204             //触发原生dom节点的点击事件
205             $(targets[targetIndex]).fire('click');
206             radios.each(function(value, key) {
207                 value.removeClass(selectedClass).removeClass(hoverClass);
208             })
209             radio.addClass(selectedClass);
210         },
211         /**
212          * 判断是否处于禁用状态
213          * @param  {HTMLElement | KISSY Node | String}  原生的dom节点，Nodelist，或者是选择器字符串
214          * @return {Boolean}
215          */
216         _isDisabled: function(target) {
217             var protoDisabled = $(target).attr('disabled'),
218                 modifyDisabled = $(target).attr('ks-radio-disabled');
219             return protoDisabled === 'disabled' || modifyDisabled === 'disabled';
220         },
221         /**
222          * 判断是否处于禁用状态
223          * @param  {HTMLElement | KISSY Node | String}  原生的dom节点，Nodelist，或者是选择器字符串
224          * @return {Boolean}
225          */
226         _isSelected: function(target) {
227             var protoselected = $(target).prop('checked'),
228                 hasselectedClass = $(target).hasClass(this.get('cls').selected);
229             return protoselected || hasselectedClass;
230         },
231         /**
232          * 根据索引禁用单个radio
233          * @param {Number} targetElement 数组radio的索引
234          */
235         setDisabled: function(targetElement) {
236             var self = this,
237                 radio = self.get('radio'),
238                 targets = self.get('target'),
239                 target, getClass = this.get('cls'),
240                 selectedClass = getClass.selected,
241                 disabledClass = getClass.disabled,
242                 hoverClass = getClass.hover;
243             //如果传递的是数字索引
244             if (typeof targetElement === 'number') {
245                 radio = $(radio[targetElement]);
246                 target = $(targets[targetElement]);
247                 radio.attr('ks-radio-disabled', 'disabled').removeClass(selectedClass + ' ' + hoverClass).addClass(disabledClass);
248                 target.attr('disabled', 'disabled');
249             }
250             radio.removeAttr('tabindex');
251             return self;
252         },
253         /**
254          * 根据索引恢复单个radio
255          * @param {Number} targetElement 数组radio的索引
256          */
257         setAvailabe: function(targetElement) {
258             var self = this,
259                 radio = self.get('radio'),
260                 targets = self.get('target'),
261                 target, disabledClass = this.get('cls').disabled;
262             //如果传递的是数字索引
263             if (S.isNumber(targetElement)) {
264                 radio = $(radio[targetElement]);
265                 target = $(targets[targetElement]);
266                 radio.removeAttr('ks-radio-disabled', 'disabled').removeClass(disabledClass);
267                 target.removeAttr('disabled', 'disabled');
268             }
269             radio.attr('tabindex', '0');
270             return self;
271         },
272         /**
273          * 获取所有选中的radio索引
274          * @return {Array} 选中的radio索引数组集合
275          */
276         getSelected: function() {
277             var self = this,
278                 target = self.get('target'),
279                 value;
280             for (i = 0, len = target.length; i < len; i++) {
281                 value = $(target[i]);
282                 if (self._isDisabled(value)) {
283                     continue;
284                 }
285                 if (self._isSelected(value)) {
286                     return i;
287                 }
288             }
289             return null;
290         }
291     }, {
292         ATTRS: /** @lends radio.prototype*/
293         {
294             /**
295              * 配置的目标,选择器的字符串
296              * @type {String}
297              */
298             target: {
299                 value: '',
300                 setter: function(v) {
301                     return $(v);
302                 },
303                 getter: function(v) {
304                     return $(v);
305                 }
306             },
307             /**
308              * 美化后的radio数组
309              * @type {Array}
310              * @default []
311              */
312             radio: {
313                 value: []
314             },
315             /**
316              * 一组样式名
317              * @type {Object}
318              * @default cls:{init: 'ks-radio',selected: 'ks-radio-selected',disabled: 'ks-radio-disabled',hover: 'ks-radio-hover'}
319              */
320             cls: {
321                 value: {
322                     init: 'ks-radio',
323                     selected: 'ks-radio-selected',
324                     disabled: 'ks-radio-disabled',
325                     hover: 'ks-radio-hover'
326                 }
327             },
328             /**
329              * css模块路径
330              * @default gallery/form/1.2/radio/themes/default/style2.css
331              */
332             cssUrl: {
333                 value: 'gallery/form/1.2/radio/themes/default/style.css'
334             },
335             /**
336              * 是否需要label for的对应
337              * @default false
338              */
339             hasLabel: {
340                 value: false
341             },
342             /**
343              * 通过radio查找对应label的函数
344              * @default undefined
345              * @type {Function}
346              */
347             getLabelFunc: {
348                 value: undefined,
349                 setter: function(v) {
350                     return v;
351                 },
352                 getter: function(v) {
353                     return v;
354                 }
355             },
356             /**
357              * 无障碍，建立在label的基础上,查找label里面的innerHTML
358              * @default false
359              */
360             accessible: {
361                 value: false
362             }
363         }
364     })
365     return Radio;
366 }, {
367     requires: ['node', 'base']
368 });
