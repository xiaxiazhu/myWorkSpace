1 /**
  2  * @fileoverview 基于html属性的规则抽象类
  3  * @author czy88840616 <czy88840616@gmail.com>
  4  *
  5  */
  6 KISSY.add('gallery/form/1.2/auth/rule/html/propertyRule', function(S, BaseRule, Utils, undefined) {
  7 
  8     /**
  9      * 属性规则
 10      *
 11      * @param {String} ruleName
 12      * @param {Function} ruleBody
 13      * @param {Object} rule params and msg
 14      * @constructor
 15      */
 16     var ProPertyRule = function() {
 17         var self = this;
 18         var args = [].slice.call(arguments);
 19         if(!args.length) {
 20             S.log('please use a name to define property');
 21             return;
 22         }
 23         self._name = args[0];
 24         var cfg = args[2]||{args:[]};
 25 
 26         self._initArgs = cfg.args;
 27         //_propertyValue和_el如果要修改必须通过属性的修改
 28         self._propertyValue = cfg.propertyValue;
 29         self._el = cfg.el;
 30         ProPertyRule.superclass.constructor.apply(self, args.slice(1));
 31     };
 32 
 33     S.extend(ProPertyRule, BaseRule, {
 34         validate:function () {
 35             var self = this;
 36             if(S.isUndefined(arguments[0])) {
 37                 return ProPertyRule.superclass.validate.apply(this, [self._propertyValue, Utils.getValue(self._el)].concat(self._initArgs));
 38             } else {
 39                 //bugfix for no args input
 40                 var args = [].slice.call(arguments);
 41                 //一旦传入过值之后，表示复写初始化的参数
 42                 self._initArgs = args;
 43                 //将属性的value作为第一个参数传进去，将当前元素的值当成第二个参数传入
 44                 return ProPertyRule.superclass.validate.apply(this, [self._propertyValue, Utils.getValue(self._el)].concat(args));
 45             }
 46         }
 47     });
 48 
 49     return ProPertyRule;
 50 }, {
 51     requires:[
 52         '../base',
 53         '../../utils'
 54     ]
 55 });
