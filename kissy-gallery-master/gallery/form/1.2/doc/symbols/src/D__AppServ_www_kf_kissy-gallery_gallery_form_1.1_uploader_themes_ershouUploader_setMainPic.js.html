  2  * @fileoverview 设置为主图功能，本来想作为插件去写，但是发现这么简单的功能不适合做插件，做成插件反而复杂了。

  7 KISSY.add('gallery/form/1.2/uploader/themes/ershouUploader/setMainPic', function(S, Node){
  8 	
  9 	var $ = Node.all,
 10 		LOG_PRE = '[LineQueue: setMainPic] ';
 11 	
 12 	function SetMainPic(mainPicInput, queueContainer){
 13 		var self = this,
 14 			mainPicInput = $(mainPicInput),
 15 			queueContainer = $(queueContainer);
 16 		// config = S.mix(_config, config);
 17 		if(!mainPicInput || mainPicInput.length <= 0){
 18 			S.log(LOG_PRE + 'cannot find mainPicInput, SetMainPic function disabled.');
 19 			return false;
 20 		}
 21 		if(!queueContainer || queueContainer.length <= 0){
 22 			S.log(LOG_PRE + 'cannot find queue container');
 23 			return false;
 24 		}
 25 		self.queueContainer = queueContainer;
 26 		self.input = mainPicInput;
 27 	}
 28 	
 29 	S.augment(SetMainPic, {
 30 		/**
 31 		 * 将队列项设置为主图
 32 		 * @param {HTMLElement|String} liElem 需要设置主图的li元素或者是主图路径
 33 		 */
 34 		setMainPic: function(liElem){
 35 			var self = this,
 36 				// container = self.container,
 37 				queueContainer = self.queueContainer,
 38 				uploadQueue = $('li', queueContainer);
 39 			if(S.isString(liElem)){
 40 				S.each(uploadQueue, function(item, index){
 41 					var url = $(item).attr('data-url');
 42 					if(url == liElem){
 43 						liElem = item;
 44 						return true;
 45 					}
 46 				});
 47 			}
 48 			var	curMainPic = self.getMainPic(),
 49 				liElem = $(liElem);
 50 			if(!liElem || liElem.length <= 0){
 51 				// var uploadQueue = $('li', queueContainer);
 52 				if(!uploadQueue[0]){
 53 					S.log(LOG_PRE + 'There is no pic. I cannot set any pic as main pic. So I will empty the main pic input.');
 54 					$(self.input).val('');
 55 					return null;
 56 				}else{
 57 					if(curMainPic.length > 0){
 58 						S.log(LOG_PRE + 'Already have a main pic. Since you do not tell me which one to set as main pic, I will do nothing.');
 59 						return curMainPic;
 60 					}else{
 61 						S.log(LOG_PRE + 'No li element specified. I will set the first pic as main pic.');
 62 						liElem = uploadQueue[0];
 63 					}
 64 				}
 65 			}
 66 			var	liWrapper = $('.J_Wrapper', liElem),
 67 				mainPicLogo = $('<span class="main-pic-logo">主图</span>'),
 68 				mainPicUrl = $(liElem).attr('data-url');
 69 			if(curMainPic.length > 0){
 70 				$(curMainPic).removeClass('main-pic');
 71 				$('.main-pic-logo', curMainPic).remove();
 72 			}
 73 			$(liElem).addClass('main-pic');
 74 			$(mainPicLogo).appendTo(liWrapper);
 75 			$(self.input).val(mainPicUrl);
 76 			S.log(LOG_PRE + 'write main pic url to :' + mainPicUrl);
 77 			return liElem;
 78 		},
 79 		
 80 		/**
 81 		 * 获取当前主图所在li
 82 		 */
 83 		getMainPic: function(){
 84 			var self = this;
 85 			return $(self.queueContainer).children('.main-pic');
 86 		},
 87 		/**
 88 		 * 获取当前主图的路径
 89 		 */
 90 		getMainPicUrl: function(){
 91 			var self = this;
 92 			return $(self.input).val();
 93 		}
 94 	});
