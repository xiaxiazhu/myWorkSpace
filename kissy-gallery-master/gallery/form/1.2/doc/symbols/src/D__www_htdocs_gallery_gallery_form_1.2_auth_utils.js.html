  6 KISSY.add('gallery/form/1.2/auth/utils', function (S, DOM, undefined) {
  7     var Utils = {
  8         toJSON:function (cfg) {
  9             cfg = cfg.replace(/'/g, '"');
 10             try {
 11                 eval("cfg=" + cfg);
 12             } catch (e) {
 13                 S.log('data-valid json is invalid');
 14             }
 15             return cfg;
 16         },
 17         guid:function () {
 18             return 'AUTH_' + S.guid();
 19         },
 20         getEvent: function(els){
 21             var event = 'blur',
 22                 type = DOM.attr(els, 'type');
 23             switch (type) {
 24                 case "select-multiple":
 25                 case "radio":
 26                 case "checkbox":
 27                     event='click';
 28                     break;
 29                 default:
 30                     event = 'blur';
 31             }
 32             return event;
 33         },
 34         getValue:function(els) {
 35             var val = [],
 36                 type = DOM.attr(els, 'type');
 37             switch (type) {
 38                 case "select-multiple":
 39                     S.each(els.options, function(el) {
 40                         if (el.selected)val.push(el.value);
 41                     });
 42                     break;
 43                 case "radio":
 44                 case "checkbox":
 45                     S.each(els, function(el) {
 46                         if (el.checked)val.push(el.value);
 47                     });
 48                     break;
 49                 default:
 50                     val = DOM.val(els);
 51             }
 52             return val;
 53         }
 54     };
 55 
 56     return Utils;
 57 },{
 58     requires:[
 59         'dom'
 60     ]
 61 });
