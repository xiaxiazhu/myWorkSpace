  6 KISSY.add('gallery/form/1.2/auth/base', function (S, JSON, Base, Field,
  7                                                   Factory, Utils, undefined) {
  8 
  9     /**
 10      * 默认配置
 11      * @type {Object}
 12      */
 13     var defaultConfig = {
 14         autoBind:true
 15     };
 16 
 17     var AUTH_MODE = {
 18         FORM:'form',
 19         OBJECT:'object'
 20     };
 21 
 22     var Auth = function (el, config) {
 23         var form = S.get(el),
 24             self = this;
 25         if (!form) {
 26             S.log('[Auth]:form element not exist');
 27         } else {
 28             self.mode = AUTH_MODE.FORM;
 29             self._init(form, S.merge(defaultConfig, config));
 30         }
 31 
 32         Auth.superclass.constructor.call(self);
 33 
 34         return self;
 35     };
 36 
 37     S.extend(Auth, Base, {
 38         _init:function (el, config) {
 39             var forms = el.elements,
 40                 self = this;
 41 
 42             //init
 43             self._storages = {};
 44 
 45             //如果是form模式，需要屏蔽html5本身的校验
 46             if(self.mode === AUTH_MODE.FORM) {
 47                 S.one(el).attr('novalidate', 'novalidate');
 48             }
 49 
 50             if (forms && forms.length) {
 51                 S.each(forms, function (el, idx) {
 52                     var filedConfig = S.merge(config, {event:config.autoBind ? Utils.getEvent(el):'none'});
 53                     var f = new Field(el, filedConfig);
 54                     f.addTarget(self);
 55                     f.publish('validate', {
 56                         bubble:1
 57                     });
 58 
 59                     self.add(f);
 60                 });
 61             }
 62 
 63             //save config
 64             self.AuthConfig = config;
 65         },
 66         add:function (field, config) {
 67             var el, key, self = this;
 68 
 69             if (field instanceof Field) {
 70                 //add field
 71                 el = field.get('el');
 72                 key = S.one(el).attr('id') || S.one(el).attr('name');
 73                 self._storages[key || Utils.guid()] = field;
 74             } else {
 75                 //add html element
 76                 el = S.one(field);
 77                 if (el) {
 78                     key = S.one(el).attr('id') || S.one(el).attr('name');
 79                     var filedConfig = S.merge(self.AuthConfig, {event:self.AuthConfig.autoBind ? Utils.getEvent(el):'none'}, config);
 80                     self._storages[key || Utils.guid()] = new Field(el, filedConfig);
 81                 }
 82             }
 83 
 84             return self;
 85         },
 86         getField:function (name) {
 87             return this._storages[name];
 88         },
 89         /**
 90          * 对Auth注册一个新的规则，当前上下文可用
 91          * @param name
 92          * @param rule
 93          */
 94         register:function (name, rule) {
 95             Factory.register(name, rule);
 96 
 97             return this;
 98         },
 99         validate:function (group) {
100             var self = this;
101 
102             self.fire('beforeValidate');
103 
104             var result = true;
105 
106             S.each(self._storages, function (field, idx) {
107                 var r = field.validate();
108                 result = result && r;
109             });
110 
111             self.fire('afterValidate');
112 
113             return result;
114         }
115     }, {
116         ATTRS:{
117 
118         }
119     });
120 
121     S.mix(Auth, {
122         Field:Field
123     });
124 
125     return Auth;
126 }, {
127     requires:[
128         'json',
129         'base',
130         './field/field',
131         './rule/ruleFactory',
132         './utils'
133     ]
134 });
