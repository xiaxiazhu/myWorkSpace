1 /**
  2  * @fileoverview 二手市场图片上传主题

  5 KISSY.add('gallery/form/1.2/uploader/themes/ershouUploader/index', function (S, Node, Theme, Message, SetMainPic) {
  6     var EMPTY = '', 
  7     	$ = Node.all,
  8     	LOG_PRE = '[Theme-ershou] ';
  9 
 10     /**
 11      * @name ErshouUploader
 12      * @class 二手市场图片上传主题
 13      * @constructor
 14      * @extends Theme
 15      * @requires Theme
 16      * @requires  ProgressBar
 17      * @author 紫英（橘子）<daxingplay@gmail.com>
 18      */
 19     function ErshouUploader(config) {
 20         var self = this;
 21         //调用父类构造函数
 22         ErshouUploader.superclass.constructor.call(self, config);
 23     }
 24     
 25     S.extend(ErshouUploader, Theme, /** @lends ErshouUploader.prototype*/ {
 26     	/**
 27          * 在上传组件运行完毕后执行的方法
 28          * 本例主要是用来绑定事件，初始化一些附加模块
 29          * @param {Uploader} uploader
 30          */
 31 		afterUploaderRender: function(uploader){
 32 			var self = this,
 33 				Plugins = self.get('oPlugin'),
 34 				queue = uploader.get('queue'),
 35                 button = uploader.get('button'),
 36                 auth = uploader.get('auth'),
 37                 queueTarget = self.get('queueTarget');
 38 			
 39 			// 初始化成功后将默认的input隐藏掉。这个默认的input是用来防止uploader无法初始化成功的fallback方案
 40 			var elemButtonTarget = button.get('target'),
 41 				elemTempFileInput = $('.original-file-input', elemButtonTarget);
 42 			$(elemTempFileInput).remove();
 43 			S.log(LOG_PRE + 'old input removed.');
 44 			
 45 			// 取得最大上传数量
 46 			var maxFileAllowed = 5;
 47 			if(auth){
 48             	maxFileAllowed = auth.getRule('max');
 49             }else{
 50             	S.log(LOG_PRE + 'Cannot get auth');
 51             }
 52             self.set('maxFileAllowed', maxFileAllowed);
 53             
 54             // 初始化插件+附加模块
 55 			var preview = new Plugins.preview(),
 56 				message = new Message({
 57 	            	'msgContainer': self.get('msgContainer'),
 58 	            	'successMsgCls': self.get('successMsgCls'),
 59 	            	'hintMsgCls': self.get('hintMsgCls'),
 60 	            	'errorMsgCls': self.get('errorMsgCls')
 61 	            }),
 62 	            setMainPic = new SetMainPic(self.get('mainPicInput'), self.get('queueTarget'));
 63             self.set('message', message);
 64             self.set('preview', preview);
 65             self.set('setMainPic', setMainPic);
 66             
 67             // 如果是ajax上传模式，添加一个class
 68             if(uploader.get('type') == 'ajax'){
 69             	S.log(LOG_PRE + 'advance queue');
 70             	$(self.get('queueTarget')).addClass('advance-queue');
 71             }
 72             
 73             // 设置主图
 74             $(queueTarget).delegate('click', '.J_SetMainPic', function(e){
 75             	e.preventDefault();
 76             	var setMainPicBtn = e.currentTarget,
 77             		// fileid = $(setMainPicBtn).attr('data-file-id'),
 78             		// fileIndex = queue.getFileIndex(fileid),
 79             		// file = queue.getFile(fileIndex),
 80             		curQueueItem = $(setMainPicBtn).parent('li');
 81         		setMainPic.setMainPic(curQueueItem);
 82             });
 83             
 84             // 删除图片
 85             $(queueTarget).delegate('click', '.J_DeleltePic', function(e){
 86             	e.preventDefault();
 87             	var delBtn = e.currentTarget,
 88             		fileid = $(delBtn).attr('data-file-id');
 89         		queue.remove(fileid);
 90             });
 91             
 92             return true;
 93             
 94             queue.on('restore', function(e){
 95             	var curMainPicUrl = setMainPic.getMainPicUrl(),
 96             		successFiles = queue.getFiles('success'),
 97             		queueLength = successFiles ? parseInt(successFiles.length, '10') : 0 + parseInt(e.filesData.length, '10');
 98             	setMainPic.setMainPic(curMainPicUrl);
 99             	if(queueLength){
100         			message.send(S.substitute(leftMsg, {
101 	        			'left': maxFileAllowed[0] - queueLength
102 	        		}), 'hint');
103         		}
104             });
105             
106             queue.on('add',function(ev){
107             	var elemImg = $('.J_ItemPic', ev.target),
108             		successFiles = queue.getFiles('success');
109         		preview.preview(ev.file.input, elemImg);
110         		S.log(LOG_PRE + 'preview done for file: ' + ev.file.id);
111         		if(successFiles.length + 1){
112         			message.send(S.substitute(leftMsg, {
113 	        			'left': maxFileAllowed[0] - successFiles.length - 1
114 	        		}), 'hint');
115         		}
116             });
117             
118             
119             
120             
121             queue.on('remove', function(e){
122             	var successFiles = queue.getFiles('success'),
123             		msg;
124             	setMainPic.setMainPic();
125             	if(successFiles.length){
126             		msg = S.substitute(leftMsg, {
127 	        			'left': maxFileAllowed - successFiles.length
128 	        		});
129         			
130         		}else{
131         			msg = S.substitute(defaultMsg, {
132 	        			'max': maxFileAllowed
133 	        		});
134         		}
135         		message.send(msg, 'hint');
136             });
137             
138             uploader.on('success', function(e){
139             	// debugger;
140             	// var successFiles = queue.getFiles('success'),
141             		// successFilesLength = successFiles ? successFiles.length : 0;
142         		var file = e.file,
143         			curQueueItem = file.target,
144         			serverUrl = file.sUrl;
145     			$(curQueueItem).attr('data-url', serverUrl);
146             	setMainPic.setMainPic();
147             	// message.send();
148             });
149 		},
150 		/**
151 		 * 更新文件数目
152 		 */
153 		_updateCount: function(){
154 			var self = this,
155 				queue = self.get('queue'),
156 				successFiles = queue.getFiles('success'),
157             	maxFileAllowed = self.get('maxFileAllowed')[0],
158                 message = self.get('message'),
159                 defaultMsg = self.get('defaultMsg'),
160                 leftMsg = self.get('leftMsg'),
161                 msg;
162             if(successFiles.length){
163         		msg = S.substitute(leftMsg, {
164         			'left': maxFileAllowed - successFiles.length
165         		});
166     		}else{
167     			msg = S.substitute(defaultMsg, {
168         			'max': maxFileAllowed
169         		});
170     		}
171     		message.send(msg, 'hint');
172     		S.log(LOG_PRE + 'file count updated. Message sent.')
173             // if(successFiles.length + 1){
174             	// var leftNum = maxFileAllowed[0] - successFiles.length - 1;
175             	// message.sendLeftMsg(leftNum);
176     		// }
177 		},
178 		/**
179          * 向队列添加完文件后触发的回调函数（在add事件前触发）
180          * @param {Number} index 文件索引值
181          * @param {Object} file 文件数据
182          * @return {Object} file
183          */
184 		_addCallback: function(index, file){
185 			var self = this,
186                 queue = self.get('queue'),
187                 queueItem = self._appendFileDom(file);
188             //将状态层容器写入到file数据
189             queue.updateFile(index, {
190                 target: queueItem
191             });
192             //更换文件状态为等待
193             queue.fileStatus(index, 'waiting');
194             self.displayFile(true, queueItem);
195             
196             // 更新剩余文件数量
197             self._updateCount();
198             return queue.getFile(index);
199 		},
200 		/**
201          * 文件处于等待上传状态时触发
202          * 进行图片预览
203          */
204         _waitingHandler:function (ev) {
205             var self = this,
206             	preview = self.get('preview'),
207             	file = ev.file,
208             	queueItem = file.target,
209                 elemImg = $('.J_ItemPic', queueItem);
210     		preview && preview.preview(file.input, elemImg);
211     		S.log(LOG_PRE + 'preview done for file: ' + ev.file.id);
212     		// 添加waiting状态
213     		$(queueItem).addClass('upload-waiting');
214         },
215         /**
216          * 在完成文件dom插入后执行的方法
217          * @param {Object} ev 类似{index:0,file:{},target:$target}
218          */
219         _addHandler: function(ev){
220         	S.log(LOG_PRE + 'add done.');
221         },
222         /**
223          * 文件处于开始上传状态时触发
224          */
225         _startHandler: function(ev){
226         	var self = this,
227         		file = ev.file,
228         		queueItem = file.target;
229     		$(queueItem).replaceClass('upload-waiting', 'uploading');
230         	S.log(LOG_PRE + 'start upload');
231         },
232         /**
233          * 文件处于上传成功状态时触发
234          */
235         _successHandler: function(ev){
236         	var self = this,
237         		file = ev.file,
238         		queueItem = file.target,
239         		setMainPic = self.get('setMainPic');
240     		$(queueItem).replaceClass('uploading', 'upload-done').attr('data-url', file.result.data.url);
241     		setMainPic.setMainPic();
242     		// 更新剩余文件数量
243             self._updateCount();
244         },
245         /**
246          * 删除图片后触发
247          */
248         _removeFileHandler: function(ev){
249         	var self = this,
250         		file = ev.file,
251         		queueItem = file.target;
252         	// 更新剩余文件数量
253             self._updateCount();
254             $(queueItem).remove();
255             S.log(LOG_PRE + 'file deleted.');
256         }
257     }, {
258     	ATTRS: /** @lends ErshouUploader.prototype*/ {
259     		/**
260 	         *  主题名
261 	         * @type String
262 	         * @default "ershouUploader"
263 	         */
264 	        name: {
265 	        	value: 'ershouUploader'
266         	},
267 	        /**
268 	         * css模块路径
269 	         * @type String
270 	         * @default "gallery/form/1.2/uploader/themes/ershouUploader/style.css"
271 	         */
272 	        cssUrl: { 
273 	        	value: 'gallery/form/1.2/uploader/themes/ershouUploader/style.css'
274 			},
275 	        /**
276 	         * 队列使用的模板
277 	         * @type String
278 	         */
279 	        fileTpl:{
280 	        	value: ['<li id="J_LineQueue-{id}" data-file-id="{id}" data-url="{sUrl}" data-name="{name}" data-size="{textSize}">',
281 							'<div class="J_Wrapper wrapper">',
282 								'<div class="tb-pic120">',
283 									'<a href="javascript:void(0);"><img class="J_ItemPic" src="{sUrl}" /></a>',
284 								'</div>',
285 								'<div class="pic-mask"></div>',
286 								'<div class="tips-uploading"><div class="progress-bar J_ProgressBar"><span class="progress-mask J_UploadingProgress"></span></div><p class="tips-text">上传中，请稍候</p></div>',
287 								'<div class="tips-upload-success"><span class="progress-bar"></span><p class="tips-text">上传成功！</p></div>',
288 								'<div class="tips-upload-error"><span class="progress-bar"></span><p>上传失败</p><p>请重新尝试！</p></div>',
289 								'<div class="tips-upload-waiting">等待上传，请稍候</div>',
290 								'<div class="upload-op-mask"></div>',
291 								'<div class="upload-operations">',
292 									'<a class="J_SetMainPic set-as-main" data-file-id="{id}" href="#">设为主图</a>',
293 									'<a class="J_DeleltePic del-pic" data-file-id="{id}" href="#">删除</a>',
294 								'</div>',
295 							'</div>',
296 						'</li>'].join('')
297 	        },
298 	        /**
299 	         * 需要加载的插件，需要手动实例化
300 	         * @type Array
301 	         */
302 			plugins: { 
303 				value: [
304 					'preview', 
305 					'progressBar'
306 				]
307 			},
308 			
309 			/**
310 			 * 显示错误消息的容器id
311 			 * @type String
312 			 */
313 			'msgContainer': {
314 				value: '#J_MsgBoxUpload'
315 			},
316 			/**
317 			 * 默认的提示消息
318 			 * @type String
319 			 */
320 			'defaultMsg': {
321 				value: '最多上传{max}张照片，每张图片小于5M'
322 			},
323 			/**
324 			 * 剩余多少张的消息
325 			 * @type String
326 			 */
327 			'leftMsg': {
328 				value: '还可以上传{left}张图片，每张小于5M。主图将在搜索结果中展示，请认真设置。'
329 			},
330 			/**
331 			 * 成功消息的class
332 			 * @type String
333 			 */
334 			'successMsgCls': {
335 				value: 'msg-success'
336 			},
337 			/**
338 			 * 提示消息的class
339 			 * @type String
340 			 */
341 			'hintMsgCls': {
342 				value: 'msg-hint'
343 			},
344 			/**
345 			 * 错误消息的class
346 			 * @type String
347 			 */
348 			'errorMsgCls': {
349 				value: 'msg-error'
350 			},
351 			/**
352 			 * 设置主图的input，如果不存在，则不初始化设置主图功能
353 			 * @type String
354 			 */
355 			'mainPicInput': {
356 				value: '#J_MainPicUrl'
357 			}
358     	}
359     });

361     return ErshouUploader;
