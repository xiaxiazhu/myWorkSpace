1 /**
  2  * @fileoverview 横排队列发送消息
  3  * @author 紫英（橘子）<daxingplay@gmail.com>
  4  * @date 2012-01-11
  5  */
  6 KISSY.add('gallery/form/1.1/uploader/themes/ershouUploader/message', function(S, Node){
  7 	
  8 	var $ = Node.all,
  9 		LOG_PRE = '[ershouUploader: Message] ';
 10 	
 11 	function Message(config){
 12 		var self = this;
 13 		self.config = S.mix({
 14 			msgContainer: '#J_MsgBoxUpload',
 15 			successMsgCls: 'msg-success',
 16 			hintMsgCls: 'msg-hint',
 17 			errorMsgCls: 'msg-error'
 18 		}, config);
 19 		// Message.superclass.constructor.call(self, config);
 20 		S.log(LOG_PRE + 'Constructed');
 21 	}
 22 	
 23 	S.augment(Message, {
 24 		
 25 		/**
 26 		 * 向msg容器发送消息
 27 		 */
 28 		send: function(msg, type){
 29 			var self = this;
 30 			if(!msg){
 31 				S.log(LOG_PRE + 'You did not tell me what to show.');
 32 				return false;
 33 			}
 34 			var msgBox = self.config.msgContainer,
 35 				newClsName = self.config[type + 'MsgCls'],
 36 				successCls = self.config.successMsgCls,
 37 				hintCls = self.config.hintMsgCls,
 38 				errorCls = self.config.errorMsgCls;
 39 			if(msgBox){
 40 				switch(type){
 41 					case 'success':
 42 					case 'hint':
 43 					case 'error':
 44 						$(msgBox).html(msg);
 45 						$(msgBox).replaceClass([successCls, hintCls, errorCls].join(' '), newClsName);
 46 						return true;
 47 						break;
 48 					default:
 49 						S.log(LOG_PRE + 'type error');
 50 						return false;
 51 						break;
 52 				}
 53 			}
 54 		}
 55 		
 56 	});
 57 	
 58 	return Message;
 59 	
 60 }, {
 61 	requires: [
 62 		'node'
 63 	]
 64 });
