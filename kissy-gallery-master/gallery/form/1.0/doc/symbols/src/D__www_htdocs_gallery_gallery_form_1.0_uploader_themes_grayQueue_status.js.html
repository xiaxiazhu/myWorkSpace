1 KISSY.add('gallery/form/1.0/uploader/themes/grayQueue/status',function(S, Node,ProgressBar, StatusBase) {
  2     var EMPTY = '',$ = Node.all;
  3     
  4     /**
  5      * @name GrayQueueStatus
  6      * @class 状态类
  7      * @constructor
  8      * @extends Base
  9      * @requires Node
 10      */
 11     function Status(target, config) {
 12         var self = this;
 13         //调用父类构造函数
 14         Status.superclass.constructor.call(self,target, config);
 15         self.set('target', $(target));
 16     }
 17     Status.type = StatusBase.type;
 18     S.extend(Status, StatusBase, /** @lends Status.prototype*/{
 19         /**
 20          * 等待上传时状态层内容
 21          */
 22         _waiting : function() {
 23             var self = this, tpl = self.get('tpl'),waitingTpl = tpl.waiting,
 24                 uploader = self.get('uploader'),
 25                 queue = self.get('queue'),
 26                 file = self.get('file'),
 27                 id = file.id,
 28                 //所有文件大小
 29                 total = uploader.get('total'),
 30                 //等待上传的文件大小
 31                 size = file.size,
 32                 //刷新状态层内容
 33                 $content = self._changeDom(waitingTpl),
 34                 //删除图标元素
 35                 $del = $content.children('.J_DelFile');
 36             //点击删除图标
 37             if($del.length){
 38                 $del.on('click',function(ev){
 39                     //删除队列中的文件
 40                     queue.remove(id);
 41                     //文件总字节数需要减去该文件大小
 42                     total  -= size;
 43                     $('#J_TotalSize').text(StatusBase.convertByteSize(total));
 44                     uploader.set('total',total);
 45                 })
 46             }
 47             //不存在文件大小，直接退出
 48             if(!size) return false;
 49             //如果不存在已经加载字节数，那么设置为0
 50             if(!total){
 51                 total = size;
 52                 uploader.set('loaded',0);
 53             }else{
 54                 //总字节数加上当前文件字节数
 55                 total += size;
 56             }
 57             //改变总字节数，StatusBase.convertByteSize方法，会将345345改成kb或mb单位显示
 58             $('#J_TotalSize').text(StatusBase.convertByteSize(total));
 59             uploader.set('total',total);
 60         },
 61         /**
 62          * 开始上传后改成状态层内容
 63          */
 64         _start : function(data) {
 65             var self = this, tpl = self.get('tpl'),startTpl = tpl.start,
 66                 target = self.get('target'),
 67                 uploader = self.get('uploader'),
 68                 uploadType = uploader.get('type'),
 69                 $content;
 70             if (!S.isString(startTpl)) return false;
 71             //改变状态层内容
 72             $content = self._changeDom(startTpl);
 73             //如果是ajax异步上传，加入进度显示
 74             if (uploadType == 'ajax') {
 75                 var progressBar;
 76                 //如果不存在进度条，先初始化进度条组件
 77                 if(!uploader.get('progressBar')){
 78                     progressBar = new ProgressBar($('#J_ProgressBar'));
 79                     progressBar.render();
 80                     uploader.set('progressBar',progressBar);
 81                 }
 82                 //清零进度条
 83                 uploader.get('progressBar').set('value',0);
 84                 //将进度百分比设置为0%
 85                 var $progressNum = $content.children('.J_ProgressNum');
 86                 $progressNum.html("0%");
 87                 self.set('elProgressNum',$progressNum);
 88             }
 89             //给li增加current-upload-file样式
 90             var $parent = target.parent();
 91             $parent.addClass('current-upload-file');
 92         },
 93         /**
 94          * 正在上传时候刷新状态层的内容
 95          * @param data
 96          */
 97         _progress : function(data){
 98             var self = this,
 99                 //已经加载的字节数
100                 loaded = data.loaded,
101                 //当前文件字节总数
102                 total = data.total,
103                 //百分比
104                 val = Math.ceil(loaded/total * 100),
105                 uploader = self.get('uploader'),
106                 //进度条
107                 proccessBar = uploader.get('progressBar'),
108                 //所有文件的字节总数
109                 allFileTotal = uploader.get('total'),
110                 //所有文件已经加载的字节数
111                 allFileLoaded = uploader.get('loaded'),
112                 $elProgressNum = self.get('elProgressNum');
113             if(!$elProgressNum.length || proccessBar == EMPTY) return false;
114             $elProgressNum.html(val + '%');
115             //改变总进度显示
116             loaded += allFileLoaded;
117             val = Math.ceil(loaded/allFileTotal * 100);
118             proccessBar.set('value',val);
119             $('#J_TotalProgressNum').text(val + '%');
120         },
121         /**
122          * 文件上传成功后
123          */
124         _success : function(){
125             var self = this, tpl = self.get('tpl'),successTpl = tpl.success,
126                 //状态层容器
127                 target = self.get('target'),
128                 queue = self.get('queue'),
129                 file = self.get('file'),
130                 size = file.size,
131                 uploader = self.get('uploader'),
132                 loaded = uploader.get('loaded');
133             if (!S.isString(successTpl)) return false;
134             self._changeDom(successTpl);
135             //删除li的current-upload-file样式
136             var $parent = target.parent();
137             $parent.removeClass('current-upload-file');
138             if(!size) return false;
139             //设置所有文件已经加载的字节数
140             loaded += size;
141             uploader.set('loaded',loaded);
142         }
143     }, {ATTRS : /** @lends Status*/{
144         /**
145          * 模板
146          */
147         tpl : {value : {
148             waiting : '<div class="clearfix"><div class="f-l">0%</div><div class="f-l uploader-icon del-icon J_DelFile"></div></div>',
149             start : '<div class="clearfix"><div class="J_ProgressNum"><img class="loading" src="http://img01.taobaocdn.com/tps/i1/T1F5tVXjRfXXXXXXXX-16-16.gif" alt="loading" /></div>' +
150                 '</div> ',
151             success : '<div class="uploader-icon success-icon">100%</div>',
152             cancel : '<div>已经取消上传，<a href="#reUpload" class="J_ReUpload">点此重新上传</a> </div>',
153             error : '<div class="upload-error">{msg}<a href="#fileDel" class="J_FileDel">点此删除</a></div>'
154         }
155         }
156     }});
157     return Status;
158 }, {requires : ['node','../../plugins/progressBar/progressBar','../../queue/status']});
