1 /**
  2  * 二手模板状态处理
  3  * @author 紫英(橘子)<daxingplay@gmail.com>
  4  */
  5 KISSY.add('gallery/form/1.0/uploader/themes/lineQueue/status',function(S, Node, StatusBase, ProgressBar) {
  6     var $ = Node.all,
  7     	LOG_PRE = '[LineQueue: status] ';
  8     
  9     /**
 10      * @name LineQueueStatus
 11      * @class 状态类
 12      * @constructor
 13      * @extends Base
 14      * @requires Node
 15      */
 16     function Status(target, config) {
 17         var self = this;
 18         //调用父类构造函数
 19         Status.superclass.constructor.call(self,target, config);
 20         self.set('target', $(target));
 21     }
 22     Status.type = StatusBase.type;
 23     S.extend(Status, StatusBase, /** @lends Status.prototype*/{
 24         /**
 25          * 等待上传时状态层内容
 26          */
 27         _waiting : function() {
 28             var self = this, 
 29                 uploader = self.get('uploader'),
 30                 queue = self.get('queue'),
 31                 fileContainer = self.get('file'),
 32                 id = $(fileContainer).attr('data-file-id');
 33             //不存在文件大小，直接退出
 34             if(fileContainer.length <= 0){
 35             	S.log(LOG_PRE + 'Cannot find this file');
 36             	return false;
 37             }
 38             $(fileContainer).addClass('upload-waiting');
 39         },
 40         /**
 41          * 开始上传后改成状态层内容
 42          */
 43         _start : function(data) {
 44             var self = this, 
 45             	// tpl = self.get('tpl'),
 46             	// startTpl = tpl.start,
 47                 target = self.get('target'),
 48                 uploader = self.get('uploader'),
 49                 curQueueItem = self.get('file'),
 50                 uploadType = uploader.get('type');
 51             $(curQueueItem).replaceClass('upload-waiting', 'uploading');
 52             //如果是ajax异步上传，加入进度显示
 53             if (uploadType == 'ajax') {
 54                 var progressContainer = $('.J_ProgressBar', curQueueItem),
 55                 	progressBar = $('.J_UploadingProgress', progressContainer);
 56                 self.set('progressBar', progressBar);
 57                 // var	progressBar = new ProgressBar(progressContainer);
 58                 // progressBar.set('width', $(progressContainer).width());
 59                 // progressBar.set('tpl', '<span class="progress-mask J_UploadingProgress" style="width: {value}; "></span>')
 60                 // progressBar.render();
 61                 // progressBar.set('value', '100');
 62                 // self.set('progressBar', progressBar);
 63             }
 64         },
 65         /**
 66          * 正在上传时候刷新状态层的内容
 67          * @param data
 68          */
 69         _progress : function(data){
 70             var self = this,
 71                 //已经加载的字节数
 72                 loaded = data.loaded,
 73                 //当前文件字节总数
 74                 total = data.total,
 75                 //百分比
 76                 val = Math.ceil(loaded/total * 100),
 77                 uploader = self.get('uploader'),
 78                 // fileContainer = self.get('file'),
 79                 // id = $(fileContainer).attr('id'),
 80                 //进度条
 81                 progressBar = self.get('progressBar');
 82             if(!progressBar){
 83             	S.log(LOG_PRE + 'Cannot find progress bar');
 84             	return false;
 85             };
 86             // progressBar.set('value', (100 - val));
 87             $(progressBar).width(100 - val + '%');
 88         },
 89         /**
 90          * 文件上传成功后
 91          */
 92         _success : function(){
 93             var self = this, 
 94                 curQueueItem = self.get('file'),
 95                 uploader = self.get('uploader'),
 96                 loaded = uploader.get('loaded');
 97             $(curQueueItem).replaceClass('uploading', 'upload-success');
 98             setTimeout(function(){
 99             	$(curQueueItem).replaceClass('upload-success', 'upload-done');
100             	
101 				// D.attr(curQueueItem, 'data-url', fileUrl);
102 				// if(!mainPic){
103 					// AjaxUploader.setMainPic(curQueueItem);
104 				// }
105 			}, 1000);
106 			// S.log(uploader.get('queue'), 'dir');
107         },
108         _restore: function(){
109        		var self = this, 
110                 curQueueItem = self.get('file'),
111                 uploader = self.get('uploader'),
112                 loaded = uploader.get('loaded');
113             $(curQueueItem).replaceClass('upload-waiting', 'upload-done');
114         },
115         /**
116          * 上传失败后改成状态层内容
117          */
118         _error : function(data) {
119             if(!S.isObject(data) || !data.msg){
120                 data = {
121                 	msg : '文件上传失败！'
122             	};
123             }
124             var self = this, 
125                 curQueueItem = self.get('file'),
126                 id = $(curQueueItem).attr('data-file-id'),
127                 uploader = self.get('uploader'),
128                 queue = self.get('queue'),
129                 message = uploader.get('message');
130             message.send(data.msg, 'error');
131             $(curQueueItem).replaceClass('uploading', 'upload-error');
132             setTimeout(function(){
133             	queue.remove(id);
134 				// $(curQueueItem).remove();
135 			}, 1000);
136         }
137     }, {ATTRS : /** @lends Status*/{
138         /**
139          * 模板
140          */
141         tpl : {value : {
142             waiting : '<div class="clearfix"><div class="f-l">0%</div><div class="f-l uploader-icon del-icon J_DelFile"></div></div>',
143             start : '<div class="clearfix"><div class="J_ProgressNum"><img class="loading" src="http://img01.taobaocdn.com/tps/i1/T1F5tVXjRfXXXXXXXX-16-16.gif" alt="loading" /></div>' +
144                 '</div> ',
145             success : '<div class="uploader-icon success-icon">100%</div>',
146             cancel : '<div>已经取消上传，<a href="#reUpload" class="J_ReUpload">点此重新上传</a> </div>',
147             error : '<div class="upload-error">{msg}<a href="#fileDel" class="J_FileDel">点此删除</a></div>'
148         }
149         }
150     }});
151     return Status;
152 }, {
153 	requires : [
154 		'node',
155 		// '../../queue/progressBar',
156 		'../../queue/status',
157 		'../../plugins/progressBar/progressBar'
158 	]
159 });
