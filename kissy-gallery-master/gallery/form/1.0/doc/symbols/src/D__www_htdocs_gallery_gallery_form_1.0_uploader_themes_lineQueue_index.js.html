1 /**
  2  * @fileoverview 横排队列上传主题
  3  * @author 紫英（橘子）<daxingplay@gmail.com>
  4  * @date 2012-01-11
  5  */
  6 KISSY.add('gallery/form/1.0/uploader/themes/lineQueue/index', function(S, Node, DefaultTheme, Queue, Preview, Message, SetMainPic){
  7 	
  8 	var $ = Node.all,
  9 		LOG_PRE = '[LineQueue:index] ';
 10 	
 11 	function LineQueue(config){
 12 		var self = this;
 13 		self.set('queueTarget', config.queueTarget);
 14         //调用父类构造函数
 15         LineQueue.superclass.constructor.call(self, config);
 16 	}
 17 	
 18 	S.extend(LineQueue, DefaultTheme, /** @lends GrayQueue.prototype*/{
 19 		
 20 		_init: function(){
 21 			var self = this,
 22 				queueTarget = self.get('queueTarget'),
 23 				queue;
 24             queue = new Queue(queueTarget);
 25             self.set('queue',queue);
 26             // S.log(queue);
 27             var setMainPic = new SetMainPic(self.get('mainPicInput'), self.get('queueTarget'));
 28             self.set('setMainPic', setMainPic);
 29             queue.on('restore', function(e){
 30             	var curMainPicUrl = setMainPic.getMainPicUrl();
 31             	setMainPic.setMainPic(curMainPicUrl);
 32             });
 33             S.log(LOG_PRE + 'inited.');
 34 		},
 35 		/**
 36 		 * 父类渲染成功后模板执行的自定义方法。
 37 		 */
 38 		afterUploaderRender: function(uploader){
 39 			var self = this,
 40 				queueTarget = self.get('queueTarget'),
 41 				// elemButtonTarget = uploader.get('buttonTarget'),
 42                 queue = uploader.get('queue'),
 43                 button = uploader.get('button'),
 44                 elemButtonTarget = button.get('target'),
 45                 auth = uploader.get('auth'),
 46                 elemTempFileInput = $('.original-file-input', elemButtonTarget),
 47                 elemFileInput = button.get('fileInput'),
 48                 maxFileAllowed = 5,
 49                 defaultMsg = self.get('defaultMsg'),
 50                 leftMsg = self.get('leftMsg');
 51             
 52             if(auth){
 53             	maxFileAllowed = auth.getRule('max');
 54             }else{
 55             	S.log(LOG_PRE + 'Cannot get auth');
 56             }
 57             
 58             $(elemTempFileInput).remove();
 59             S.log(LOG_PRE + 'old input removed.');
 60             
 61             // 初始化一些附加模块+插件
 62             var preview = new Preview(),
 63             	message = new Message({
 64 	            	'msgContainer': self.get('msgContainer'),
 65 	            	'successMsgCls': self.get('successMsgCls'),
 66 	            	'hintMsgCls': self.get('hintMsgCls'),
 67 	            	'errorMsgCls': self.get('errorMsgCls')
 68 	            }),
 69 	            setMainPic = self.get('setMainPic');
 70             // message.set('msgContainer', '#J_MsgBoxUpload');
 71             uploader.set('message', message);
 72             
 73             queue.on('add',function(ev){
 74             	var elemImg = $('.J_ItemPic', ev.target),
 75             		successFiles = queue.getFiles('success');
 76         		preview.preview(ev.file.input, elemImg);
 77         		S.log(LOG_PRE + 'preview done for file: ' + ev.file.id);
 78         		if(successFiles.length + 1){
 79         			message.send(S.substitute(leftMsg, {
 80 	        			'left': maxFileAllowed - successFiles.length - 1
 81 	        		}), 'hint');
 82         		}
 83             });
 84             
 85             if(uploader.get('type') == 'ajax'){
 86             	S.log(LOG_PRE + 'advance queue');
 87             	$(self.get('queueTarget')).addClass('advance-queue');
 88             }
 89             
 90             // 删除图片
 91             $(queueTarget).delegate('click', '.J_DeleltePic', function(e){
 92             	e.preventDefault();
 93             	var delBtn = e.currentTarget,
 94             		fileid = $(delBtn).attr('data-file-id');
 95         		queue.remove(fileid);
 96             	// var target = e.target;
 97             	// if($(target).hasClass('J_DeleltePic')){
 98             		// var fileid = $(target).attr('data-file-id');
 99             		// queue.remove(fileid);
100             	// }else if($(target).hasClass('J_DeleltePic')){
101 //             		
102             	// }
103             });
104             
105             // 设置主图
106             $(queueTarget).delegate('click', '.J_SetMainPic', function(e){
107             	e.preventDefault();
108             	var setMainPicBtn = e.currentTarget,
109             		// fileid = $(setMainPicBtn).attr('data-file-id'),
110             		// fileIndex = queue.getFileIndex(fileid),
111             		// file = queue.getFile(fileIndex),
112             		curQueueItem = $(setMainPicBtn).parent('li');
113         		setMainPic.setMainPic(curQueueItem);
114             });
115             
116             queue.on('remove', function(e){
117             	var successFiles = queue.getFiles('success'),
118             		msg;
119             	setMainPic.setMainPic();
120             	if(successFiles.length){
121             		msg = S.substitute(leftMsg, {
122 	        			'left': maxFileAllowed - successFiles.length
123 	        		});
124         			
125         		}else{
126         			msg = S.substitute(defaultMsg, {
127 	        			'max': maxFileAllowed
128 	        		});
129         		}
130         		message.send(msg, 'hint');
131             });
132             
133             uploader.on('success', function(e){
134             	// debugger;
135             	// var successFiles = queue.getFiles('success'),
136             		// successFilesLength = successFiles ? successFiles.length : 0;
137         		var file = e.file,
138         			curQueueItem = file.target,
139         			serverUrl = file.sUrl;
140     			$(curQueueItem).attr('data-url', serverUrl);
141             	setMainPic.setMainPic();
142             	// message.send();
143             });
144 		}
145 	}, {
146 		ATTRS: {
147             cssUrl:{value:'gallery/form/1.0/uploader/themes/lineQueue/style.css'},
148 			// 消息容器，为空则不初始化消息
149 			'msgContainer': {
150 				value: '#J_MsgBoxUpload'
151 			},
152 			// 默认消息
153 			'defaultMsg': {
154 				value: '最多上传{max}张照片，每张图片小于5M'
155 			},
156 			// 剩余多少张的消息
157 			'leftMsg': {
158 				value: '还可以上传{left}张图片，每张小于5M。主图将在搜索结果中展示，请认真设置。'
159 			},
160 			'successMsgCls': {
161 				value: 'msg-success'
162 			},
163 			'hintMsgCls': {
164 				value: 'msg-hint'
165 			},
166 			'errorMsgCls': {
167 				value: 'msg-error'
168 			},
169 			// 设置主图的input，如果不存在，则不初始化设置主图功能
170 			'mainPicInput': {
171 				value: '#J_MainPicUrl'
172 			}
173 		}
174 	});
175 	
176 	return LineQueue;
177 	
178 }, {
179 	requires: [
180 		'node',
181 		'../default/index',
182 		'./queue',
183 		'../../plugins/preview/preview',
184 		'./message',
185 		'./setMainPic'
186 	]
187 });
