1 KISSY.add('gallery/form/1.0/uploader/themes/default/index', function (S, Node, Base, Queue) {
  2     var EMPTY = '', $ = Node.all;
  3 
  4     /**
  5      * @name DefaultTheme
  6      * @class 上传组件默认模板
  7      * @constructor
  8      * @extends Base
  9      * @requires Node
 10      */
 11     function DefaultTheme(config) {
 12         var self = this;
 13         //调用父类构造函数
 14         DefaultTheme.superclass.constructor.call(self, config);
 15         self._LoaderCss();
 16         self._init();
 17     }
 18 
 19     S.extend(DefaultTheme, Base, /** @lends DefaultTheme.prototype*/{
 20         /**
 21          * 初始化
 22          */
 23         _init:function () {
 24             var self = this, queueTarget = self.get('queueTarget'), queue,
 25                 //处理队列的配置
 26                 config = S.parseConfig(queueTarget, 'data-queue-config');
 27             queue = new Queue(queueTarget, config);
 28             self.set('queue', queue);
 29         },
 30         /**
 31          * 加载css文件
 32          */
 33         _LoaderCss : function(){
 34             var self = this,
 35                 isUseCss = self.get('isUseCss'),
 36                 cssUrl = self.get('cssUrl');
 37             //加载css文件
 38             if(isUseCss){
 39                 S.use(cssUrl,function(){ });
 40             }
 41         },
 42         /**
 43          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 44          * @param {Uploader} uploader
 45          */
 46         afterUploaderRender:function (uploader) {
 47             var self = this,$uploadFiles = $(self.get('elUploadFiles'));
 48 
 49             //存在上传按钮，用户点击该按钮上传文件
 50             if($uploadFiles.length){
 51                 $uploadFiles.on('click',function(){
 52                     uploader.uploadFiles();
 53                 });
 54             }
 55         }
 56     }, {ATTRS:/** @lends DefaultTheme*/{
 57         /**
 58          * 是否引用css文件
 59          */
 60         isUseCss : {value:true},
 61         /**
 62          * css模块路径
 63          */
 64         cssUrl:{value:'gallery/form/1.0/uploader/themes/default/style.css'},
 65         /**
 66          * 上传按钮（上传所有等待的文件）
 67          */
 68         elUploadFiles : {value:EMPTY},
 69         queueTarget:{value:EMPTY},
 70         queue:{value:EMPTY}
 71     }});
 72     return DefaultTheme;
 73 }, {requires:['node', 'base', '../../queue/base']});
