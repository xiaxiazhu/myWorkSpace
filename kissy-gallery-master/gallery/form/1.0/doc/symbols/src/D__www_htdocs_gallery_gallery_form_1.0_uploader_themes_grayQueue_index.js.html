1 KISSY.add('gallery/form/1.0/uploader/themes/grayQueue/index',function(S, Node, DefaultTheme,Queue) {
  2     var EMPTY = '',$ = Node.all;
  3 
  4     /**
  5      * @name GrayQueue
  6      * @class 上传组件灰色模板
  7      * @constructor
  8      * @extends Base
  9      * @requires Node
 10      */
 11     function GrayQueue(config) {
 12         var self = this;
 13         //调用父类构造函数
 14         GrayQueue.superclass.constructor.call(self, config);
 15     }
 16     S.extend(GrayQueue, DefaultTheme, /** @lends GrayQueue.prototype*/{
 17         /**
 18          * 初始化
 19          */
 20         _init : function() {
 21             var self = this,queueTarget = self.get('queueTarget'),queue;
 22             queue = new Queue(queueTarget);
 23             self.set('queue',queue);
 24         },
 25         /**
 26          * 在上传组件运行完毕后执行的方法（对上传组件所有的控制都应该在这个函数内）
 27          * @param {Uploader} uploader
 28          */
 29         afterUploaderRender : function(uploader){
 30             var self = this,
 31                 queue = uploader.get('queue'),
 32                 //开始上传按钮
 33                 $startUpload = $(self.get('elStartUpload')),
 34                 //总进度数容器
 35                 $totalProgressNum = $(self.get('elTotalProgressNum')),
 36                 //上传按钮不可用时的样式名
 37                 startUploadDisabledCls = self.get('startUploadDisabledCls');
 38             //监听队列的添加文件后事件
 39             queue.on('add',function(ev){
 40                 $startUpload.removeClass( startUploadDisabledCls);
 41             });
 42             //全部上传完成后触发
 43             uploader.on('uploadAll',function(){
 44                 //进度条
 45                 var progressBar = uploader.get('progressBar');
 46                 //强制进度到100%，防止部分上传只能到99%的问题
 47                 progressBar.set('value',100);
 48                 progressBar.hide();
 49                 $totalProgressNum.text('100%');
 50             });
 51             //点击开始上传的按钮
 52             $startUpload.on('click',function(ev){
 53                 ev.preventDefault();
 54                 //如果不是禁用状态，上传所有等待中的文件
 55                 if(!$startUpload.hasClass( startUploadDisabledCls)){
 56                     var progressBar = uploader.get('progressBar');
 57                     if(progressBar) progressBar.show();
 58                     uploader.uploadAll();
 59                 }
 60             })
 61         }
 62     }, {ATTRS : /** @lends GrayQueue*/{
 63         elStartUpload : {value : '#J_StartUpload'},
 64         startUploadDisabledCls : {value : 'start-upload-disabled'},
 65         elTotalProgressNum : {value : '#J_TotalProgressNum'}
 66     }});
 67     return GrayQueue;
 68 }, {requires : ['node','../default/index','./queue','./style.css']});
