1 /**
  2  * @fileoverview 文件改变状态后改变状态元素的内容
  3  * @author 剑平（明河）<minghe36@126.com>,紫英<daxingplay@gmail.com>
  4  **/
  5 KISSY.add('gallery/form/1.0/uploader/queue/status',function(S, Node, Base,ProgressBar) {
  6     var EMPTY = '',$ = Node.all,LOG_PREFIX = '[queue-status]:';
  7 
  8     /**
  9      * @name Status
 10      * @class 文件改变状态后改变状态元素的内容
 11      * @constructor
 12      * @extends Base
 13      * @requires Node
 14      * @param {String} target 目标元素钩子
 15      * @param {Object} config 配置
 16      */
 17     function Status(target, config) {
 18         var self = this;
 19         config = S.merge({target:$(target)}, config);
 20         //调用父类构造函数
 21         Status.superclass.constructor.call(self, config);
 22     }
 23 
 24     S.mix(Status, /** @lends Status.prototype*/{
 25         /**
 26          * 文件的状态类型
 27          */
 28         type : {
 29             WAITING : 'waiting',
 30             START : 'start',
 31             PROGRESS : 'progress',
 32             SUCCESS : 'success',
 33             CANCEL : 'cancel',
 34             ERROR : 'error',
 35             RESTORE: 'restore'
 36         },
 37         /**
 38          * 转换文件大小字节数
 39          * @param {Number} bytes 文件大小字节数
 40          * @return {String} 文件大小
 41          */
 42         convertByteSize : function(bytes){
 43             var i = -1;
 44             do {
 45                 bytes = bytes / 1024;
 46                 i++;
 47             } while (bytes > 99);
 48             return Math.max(bytes, 0.1).toFixed(1) + ['kB', 'MB', 'GB', 'TB', 'PB', 'EB'][i];
 49         }
 50     });
 51     //继承于Base，属性getter和setter委托于Base处理
 52     S.extend(Status, Base, /** @lends Status.prototype*/{
 53         /**
 54          * 改变状态，调用对应的状态函数
 55          * @param {String} status 状态名
 56          * @param {Object} args 传递给状态函数的参数
 57          * @return {Status}
 58          */
 59         change : function(status,args){
 60             if (!S.isString(status)) return false;
 61             var self = this,method;
 62             if (!self.isSupport(status)) {
 63                 S.log(LOG_PREFIX + 'status参数为' + status + '，不支持的状态类型');
 64                 return false;
 65             }
 66             if(!args) args = {};
 67             method = self['_' + status];
 68             //改变状态层内容
 69             method && method.call(self,args);
 70             self.set('curType',status);
 71             return self;
 72         },
 73         /**
 74          * 判断是不是允许的状态类型
 75          * @param {String} status
 76          * @return {Boolean}
 77          */
 78         isSupport : function(status) {
 79             if (!S.isString(status)) return false;
 80             var type = Status.type,b = false;
 81             S.each(type, function(v) {
 82                 if (status == v) {
 83                     return b = true;
 84                 }
 85             });
 86             return b;
 87         },
 88         /**
 89          * 改变状态层的DOM内容
 90          * @return {NodeList} 内容层
 91          */
 92         _changeDom : function(content) {
 93             var self = this,$target = self.get('target'),$content;
 94             $target.html(EMPTY);
 95             $content = $(content).appendTo($target);
 96             return $content;
 97         },
 98         /**
 99          * 等待上传时状态层内容
100          */
101         _waiting : function() {
102             var self = this, tpl = self.get('tpl'),waitingTpl = tpl.waiting,
103                 uploader = self.get('uploader'),
104                 queue = self.get('queue'),
105                 //文件id
106                 file = self.get('file'),id = file.id,
107                 index = queue.getFileIndex(id),
108                 $content = self._changeDom(waitingTpl),
109                 $upload = $content.children('.J_Upload');
110             $upload.on('click',function(ev){
111                 ev.preventDefault();
112                 if (!S.isObject(uploader)) return false;
113                 uploader.upload(index);
114             });
115         },
116         /**
117          * 开始上传后改成状态层内容
118          */
119         _start : function(data) {
120             var self = this, tpl = self.get('tpl'),startTpl = tpl.start,
121                 target = self.get('target'),
122                 uploader = self.get('uploader'),
123                 uploadType = uploader.get('type'),
124                 $content,$cancel;
125             if (!S.isString(startTpl)) return false;
126             $content = self._changeDom(startTpl);
127             //取消链接
128             $cancel = $content.children('.J_UploadCancel');
129             $cancel.on('click', function(ev) {
130                 ev.preventDefault();
131                 if (!S.isObject(uploader)) return false;
132                 uploader.cancel();
133             });
134             //如果是ajax或flash异步上传，加入进度条
135             if(uploadType == 'ajax' || uploadType == 'flash'){
136                 var $progressBar = $content.children('.J_ProgressBar');
137                 var progressBar = new ProgressBar($progressBar);
138                 progressBar.render();
139                 self.set('progressBar',progressBar);
140             }
141             var $parent = target.parent();
142             $parent.addClass('current-upload-file');
143         },
144         /**
145          * 正在上传时候刷新状态层的内容
146          * @param data
147          */
148         _progress : function(data){
149             var self = this,loaded = data.loaded,total = data.total,
150                 val = Math.ceil(loaded/total) * 100,
151                 progressBar = self.get('progressBar');
152             if(!progressBar) return false;
153             progressBar.set('value',val);
154         },
155         /**
156          * 成功上传后改成状态层内容
157          */
158         _success : function() {
159             var self = this, tpl = self.get('tpl'),successTpl = tpl.success,
160                 target = self.get('target'),
161                 queue = self.get('queue'),
162                 file = self.get('file'),id = file.id,
163                 progressBar = self.get('progressBar'),
164                 $target = self.get('target'),
165                 $content,$progressBar,
166                 $del;
167             if (!S.isString(successTpl)) return false;
168             //设置为100%进度
169             S.isObject(progressBar) && progressBar.set('value',100);
170             S.later(function(){
171                 //拷贝进度条
172                 if(S.isObject(progressBar)){
173                    var $wrapper =$target.children();
174                    $progressBar = $wrapper.children('.J_ProgressBar').clone(true);
175                 }
176                 //改变状态层的内容
177                 $content = self._changeDom(successTpl);
178                 //将进度条插入到状态层
179                 if($progressBar) $content.prepend($progressBar);
180                 $del = $content.children('.J_FileDel');
181                 //点击删除
182                 $del.on('click', function(ev) {
183                     ev.preventDefault();
184                     //删除队列中的文件
185                     queue.remove(id);
186                 });
187             },300);
188             var $parent = target.parent();
189             $parent.removeClass('current-upload-file');
190         },
191         /**
192          * 文件恢复到队列中
193          */
194         _restore: function(){
195             this._success();
196         },
197         /**
198          * 取消上传后改成状态层内容
199          */
200         _cancel : function() {
201             var self = this, tpl = self.get('tpl'),cancelTpl = tpl.cancel,
202                 uploader = self.get('uploader'),
203                 $content = self._changeDom(cancelTpl),
204                 $reUpload = $content.children('.J_ReUpload'),
205                 //文件id
206                 file = self.get('file'),id = file.id;
207             //点击重新上传链接
208             $reUpload.on('click', function(ev) {
209                 ev.preventDefault();
210                 if (!S.isObject(uploader)) return false;
211                 uploader.upload(id);
212             });
213         },
214         /**
215          * 上传失败后改成状态层内容
216          */
217         _error : function(data) {
218             if(!S.isObject(data)){
219                 data = {msg : '文件上传失败！'};
220             }
221             var self = this, tpl = self.get('tpl'),errorTpl = tpl.error,
222                 html = S.substitute(errorTpl,data),
223                 $content = self._changeDom(html),
224                 $del = $content.children('.J_FileDel'),
225                 queue = self.get('queue'),
226                 //文件id
227                 file = self.get('file'),id = file.id;
228             //点击重新上传链接
229             $del.on('click', function(ev) {
230                 ev.preventDefault();
231                 //删除队列中的文件
232                 queue.remove(id);
233             });
234         }
235     }, {ATTRS : /** @lends Status*/{
236         /**
237          * 状态改变时改变的元素层
238          */
239         target : {value : EMPTY},
240         /**
241          * 模板
242          */
243         tpl : {value : {
244             waiting : '<div class="waiting-status">等待上传，<a href="#Upload" class="J_Upload">点此上传</a> </div>',
245             start : '<div class="start-status clearfix">' +
246                         '<div class="f-l  J_ProgressBar uploader-progress"><img class="loading" src="http://img01.taobaocdn.com/tps/i1/T1F5tVXjRfXXXXXXXX-16-16.gif" alt="loading" /></div>' +
247                         ' <a class="f-l J_UploadCancel upload-cancel" href="#uploadCancel">取消</a>' +
248                     '</div> ',
249             success : ' <div class="success-status"><a href="#fileDel" class="J_FileDel">删除</a></div>',
250             cancel : '<div class="cancel-status">已经取消上传，<a href="#reUpload" class="J_ReUpload">点此重新上传</a> </div>',
251             error : '<div class="error-status upload-error">{msg}<a href="#fileDel" class="J_FileDel">删除</a></div>'
252         } },
253         /**
254          * 队列实例
255          */
256         queue : {value : EMPTY},
257         /**
258          * 上传组件的实例
259          */
260         uploader : {value : EMPTY},
261         /**
262          * 文件对象
263          */
264         file : {value : {}},
265         /**
266          * 当前状态类型
267          */
268         curType : { value : EMPTY },
269         //进度条ProgressBar的实例，iframe上传时并不存在
270         progressBar : {value : EMPTY}
271     }});
272     return Status;
273 }, {requires : ['node','base','../plugins/progressBar/progressBar']});
